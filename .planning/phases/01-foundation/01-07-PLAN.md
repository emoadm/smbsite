## Objective

Implement the four canonical Phase 1 Server Actions: `register`, `requestOtp`, `verifyOtp`, `logout`. Each follows the canonical signature from PATTERNS.md (`(prevState, formData) → { ok, error?, fieldErrors? }`), validates with Zod + Bulgarian error messages via `zod-i18n-map`, applies the anti-abuse stack from plan 1.06 in the documented order, persists hashed OTPs via `persistHashedOtp` from plan 1.05, and **enqueues** every email via the queue interface defined here (the actual BullMQ wiring lands in plan 1.10). This plan also creates the consent-row writer (D-13: append-only INSERTs into `consents` per checkbox).

## Truths

- D-08: Registration captures `full_name`, `email`, `sector`, `role`, plus 4 consent rows.
- D-09 / D-10: Sector and role values are CLOSED enums — Zod rejects anything else.
- D-12: 4 consent checkboxes; #1 (privacy_terms) and #2 (cookies) REQUIRED to submit; #3 (newsletter) and #4 (political_opinion) OPTIONAL but always recorded as INSERT (granted: true|false).
- D-13: Consent rows are append-only — only INSERTs allowed; never UPDATE.
- D-19: NEVER send email synchronously. Always enqueue. Server Action returns ≤200ms regardless of Brevo latency.
- D-07 / Pattern 5: Anti-abuse order is rate-limit → Turnstile → disposable-email → business logic. Fail closed on any failure.
- RESEARCH § Honeypot Field + Time-to-Submit (lines 658-664): hidden `website` field; reject if non-empty (silent success); reject submissions with `form_rendered_at` < 3 seconds ago.
- D-06: Disposable rejection message is generic — `auth.register.invalidEmail` ("Този имейл не може да се използва. Опитай с друг адрес.").
- The queue interface in `src/lib/email/queue.ts` here exposes `addEmailJob({ to, otpCode, expiresAt, kind })` with a TODO body — plan 1.10 implements the BullMQ side.

    import crypto from 'node:crypto';

    /**
     * RESEARCH § Honeypot Field + Time-to-Submit.
     * - `website` hidden field MUST be empty when humans submit (CSS-hides the field).
     * - `form_rendered_at` is a signed timestamp; if delta < 3000ms → bot.
     */
    const MIN_DWELL_MS = 3_000;
    const TOKEN_TTL_MS = 30 * 60 * 1000; // 30 minutes

    function key(): string {
      return process.env.AUTH_SECRET ?? 'dev-secret';
    }

    /** Server-render: include this in the form as `<input type="hidden" name="formStamp" value={signFormStamp()}>`. */
    export function signFormStamp(now = Date.now()): string {
      const sig = crypto.createHmac('sha256', key()).update(String(now)).digest('hex').slice(0, 16);
      return `${now}.${sig}`;
    }

    /** Returns 'ok' when human (>=3s dwell, valid sig); 'bot' when fast or forged; 'expired' when TTL exceeded. */
    export function checkFormStamp(stamp: string | null | undefined, now = Date.now()): 'ok' | 'bot' | 'expired' {
      if (!stamp || !stamp.includes('.')) return 'bot';
      const [tsRaw, sig] = stamp.split('.');
      const ts = Number(tsRaw);
      if (!Number.isFinite(ts)) return 'bot';
      const expectedSig = crypto.createHmac('sha256', key()).update(String(ts)).digest('hex').slice(0, 16);
      if (sig !== expectedSig) return 'bot';
      const dwell = now - ts;
      if (dwell < MIN_DWELL_MS) return 'bot';
      if (dwell > TOKEN_TTL_MS) return 'expired';
      return 'ok';
    }

    /** Honeypot field name + check. Bots tend to fill all visible inputs. */
    export const HONEYPOT_FIELD = 'website';
    export function isHoneypotTriggered(value: FormDataEntryValue | null): boolean {
      return typeof value === 'string' && value.trim().length > 0;
    }

    const SectorEnum = z.enum(['it','trade','production','services','other']);
    // D-10: closed role enum
    const RoleEnum   = z.enum(['owner','manager','employee','other']);

    const RegistrationSchema = z.object({
      full_name: z.string().min(2).max(120),
      email: z.string().trim().toLowerCase().email().refine((e) => !isDisposable(e), {

      consent_political:     z.union([z.literal('on'), z.literal('')]).optional(),
      'cf-turnstile-response': z.string().min(1),
      formStamp: z.string().min(1),
    });

    export type ActionState =
      | { ok: true; nextHref: string }
      | { ok: false; error?: string; fieldErrors?: Record<string, string[]> };

    /** Today's policy version. Bumped manually when privacy/terms text changes. */
    const POLICY_VERSION = '2026-04-29';

    export async function register(_prevState: ActionState | null, formData: FormData): Promise<ActionState> {
      // 1. HONEYPOT + DWELL — silent success on bot detection (do not leak the signal)
      if (isHoneypotTriggered(formData.get(HONEYPOT_FIELD))) {
        return { ok: true, nextHref: '/auth/otp' }; // fake success
      }
      const stampStatus = checkFormStamp(formData.get('formStamp')?.toString());
      if (stampStatus !== 'ok') return { ok: true, nextHref: '/auth/otp' };

      // 2. RATE LIMIT (per IP + per /24 subnet) — D-07
      const h = await headers();
      const ip = getClientIp(h);
      if (ip) {
        try {
          const ipR = await checkRegistrationIp(ip);
          if (!ipR.success) return { ok: false, error: 'auth.register.rateLimited' };
          const subR = await checkRegistrationSubnet(getSubnet(ip));
          if (!subR.success) return { ok: false, error: 'auth.register.rateLimited' };
        } catch {
          return { ok: false, error: 'auth.register.rateLimited' }; // fail closed
        }
      }

      // 3. PARSE + VALIDATE
      const raw = Object.fromEntries(formData.entries());
      const parsed = RegistrationSchema.safeParse(raw);
      if (!parsed.success) {
        return { ok: false, fieldErrors: parsed.error.flatten().fieldErrors };
      }
      const data = parsed.data;

      // 4. TURNSTILE — server-side siteverify (AUTH-08)
      const tr = await verifyTurnstile(data['cf-turnstile-response'], ip ?? undefined);
      if (!tr.ok) return { ok: false, error: 'auth.register.captchaFailed' };

      // 5. UNIQUENESS — uniqueness on email (Drizzle unique constraint will also catch races)
      const existing = await db.select({ id: users.id }).from(users).where(eq(users.email, data.email)).limit(1);
      if (existing.length > 0) {
        // Do not reveal that the email exists — silent success steers attacker away.
        return { ok: true, nextHref: '/auth/otp' };
      }

      // 6. INSERT user + consent rows in a transaction (D-13 append-only)
      const [created] = await db.transaction(async (tx) => {
        const inserted = await tx.insert(users).values({
          email: data.email,
          name: data.full_name,
          full_name: data.full_name,
          sector: data.sector,
          role: data.role,
        }).returning({ id: users.id, email: users.email });

        const userId = inserted[0]!.id;
        await tx.insert(consents).values([
          { user_id: userId, kind: 'privacy_terms',     granted: true,  version: POLICY_VERSION },
          { user_id: userId, kind: 'cookies',           granted: true,  version: POLICY_VERSION },
          { user_id: userId, kind: 'newsletter',        granted: data.consent_newsletter === 'on', version: POLICY_VERSION },
          { user_id: userId, kind: 'political_opinion', granted: data.consent_political  === 'on', version: POLICY_VERSION },
        ]);
        return inserted;
      });

      // 7. GENERATE + PERSIST hashed OTP (Pitfall K) and ENQUEUE email (D-19)
      const code = generateOtpCode();
      await persistHashedOtp(data.email, code, 'register');
      await addEmailJob({
        to: data.email,
        kind: 'register-otp',
        otpCode: code,
        expiresAt: registrationOtpExpiry(),
        fullName: data.full_name,
      });

      return { ok: true, nextHref: `/auth/otp?email=${encodeURIComponent(data.email)}` };
    }

    'use server';
    import { headers } from 'next/headers';
    import { z } from '@/lib/zod-i18n';
    import { db } from '@/db';
    import { users } from '@/db/schema';
    import { eq } from 'drizzle-orm';
    import { generateOtpCode, loginOtpExpiry } from '@/lib/auth-utils';
    import { persistHashedOtp } from '@/lib/auth';
    import { checkLoginOtpEmail, checkLoginOtpIp } from '@/lib/rate-limit';
    import { getClientIp } from '@/lib/ip';
    import { addEmailJob } from '@/lib/email/queue';

    const Schema = z.object({
      email: z.string().trim().toLowerCase().email(),
    });

    export type RequestOtpState =
      | { ok: true; nextHref: string }
      | { ok: false; error?: string; fieldErrors?: Record<string, string[]> };

    export async function requestOtp(_prev: RequestOtpState | null, formData: FormData): Promise<RequestOtpState> {
      const parsed = Schema.safeParse(Object.fromEntries(formData.entries()));
      if (!parsed.success) return { ok: false, fieldErrors: parsed.error.flatten().fieldErrors };
      const email = parsed.data.email;

      const h = await headers();
      const ip = getClientIp(h);
      try {
        const er = await checkLoginOtpEmail(email);
        if (!er.success) return { ok: false, error: 'auth.login.rateLimited' };
        if (ip) {
          const ir = await checkLoginOtpIp(ip);
          if (!ir.success) return { ok: false, error: 'auth.login.rateLimited' };
        }
      } catch {
        return { ok: false, error: 'auth.login.rateLimited' };
      }

      // Look up user. Silent success on missing email (do not leak account existence).
      const found = await db.select({ id: users.id }).from(users).where(eq(users.email, email)).limit(1);
      if (found.length === 0) {
        return { ok: true, nextHref: `/auth/otp?email=${encodeURIComponent(email)}` };
      }

      const code = generateOtpCode();
      await persistHashedOtp(email, code, 'login');
      await addEmailJob({ to: email, kind: 'login-otp', otpCode: code, expiresAt: loginOtpExpiry() });

      return { ok: true, nextHref: `/auth/otp?email=${encodeURIComponent(email)}` };
    }

    **Step 2.** `src/app/actions/verify-otp.ts` — submit code, validate, create session, mark email verified:
    ```typescript
    'use server';
    import crypto from 'node:crypto';
    import { cookies } from 'next/headers';
    import { z } from '@/lib/zod-i18n';
    import { db } from '@/db';
    import { users, verificationTokens, sessions } from '@/db/schema';
    import { and, eq, gt } from 'drizzle-orm';
    import { hashOtp, verifyOtpHash, MAX_OTP_VERIFY_ATTEMPTS } from '@/lib/auth-utils';
    import { checkOtpVerify } from '@/lib/rate-limit';

    const Schema = z.object({
      email: z.string().trim().toLowerCase().email(),
      code: z.string().regex(/^\d{6}$/),
    });

    export type VerifyOtpState =
      | { ok: true; nextHref: string }
      | { ok: false; error: 'auth.otp.invalid' | 'auth.otp.expired' | 'auth.otp.locked' };

    export async function verifyOtp(_prev: VerifyOtpState | null, formData: FormData): Promise<VerifyOtpState> {
      const parsed = Schema.safeParse(Object.fromEntries(formData.entries()));
      if (!parsed.success) return { ok: false, error: 'auth.otp.invalid' };
      const { email, code } = parsed.data;

      // D-07: per-OTP fixed window — key the limiter on (email, hash) so each OTP gets 5 tries
      const submittedHash = hashOtp(code, email);
      try {
        const r = await checkOtpVerify(`${email}:${submittedHash}`);
        if (!r.success) return { ok: false, error: 'auth.otp.locked' };
      } catch {
        return { ok: false, error: 'auth.otp.locked' };
      }

      // Look for any matching unexpired token for this identifier.
      const now = new Date();
      const candidates = await db.select()
        .from(verificationTokens)
        .where(and(eq(verificationTokens.identifier, email), gt(verificationTokens.expires, now)));
      if (candidates.length === 0) return { ok: false, error: 'auth.otp.expired' };

      const match = candidates.find((t) => verifyOtpHash(code, email, t.token));
      if (!match) {
        // Increment attempts on the most recent token; if attempts >= MAX, delete to invalidate.
        const latest = candidates.sort((a, b) => +b.expires - +a.expires)[0]!;
        const nextAttempts = Number(latest.attempts ?? '0') + 1;
        if (nextAttempts >= MAX_OTP_VERIFY_ATTEMPTS) {
          await db.delete(verificationTokens).where(and(eq(verificationTokens.identifier, email), eq(verificationTokens.token, latest.token)));
          return { ok: false, error: 'auth.otp.locked' };
        }
        await db.update(verificationTokens).set({ attempts: String(nextAttempts) })
          .where(and(eq(verificationTokens.identifier, email), eq(verificationTokens.token, latest.token)));
        return { ok: false, error: 'auth.otp.invalid' };
      }

      // Valid — single-use: delete this token (and any others for same identifier) and mark verified.
      await db.transaction(async (tx) => {
        await tx.delete(verificationTokens).where(eq(verificationTokens.identifier, email));
        await tx.update(users).set({
          emailVerified: now,
          email_verified_at: now, // forward-compat for Phase 3 cooling period
        }).where(eq(users.email, email));
      });

      // H-3: establish session by writing the `sessions` row directly (Auth.js database strategy).
      // We do not call `signIn('email', ...)` — that would trigger a second OTP cycle because
      // Auth.js's email provider owns its own verification loop. Instead we mint the session token
      // ourselves and set the cookie that `auth()` reads (matches Drizzle adapter expectations).
      const sessionToken = crypto.randomUUID();
      const sessionExpires = new Date(Date.now() + 30 * 24 * 60 * 60 * 1000); // D-03 (30d)
      const userRow = await db.select({ id: users.id }).from(users).where(eq(users.email, email)).limit(1);
      const userId = userRow[0]!.id;
      await db.insert(sessions).values({ sessionToken, userId, expires: sessionExpires });
      const cookieJar = await cookies();
      cookieJar.set('__Secure-next-auth.session-token', sessionToken, {
        httpOnly: true, sameSite: 'lax', secure: true, path: '/', expires: sessionExpires,
      });
      return { ok: true, nextHref: '/member' };
    }
    ```
