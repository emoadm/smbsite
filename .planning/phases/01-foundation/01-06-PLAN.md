    ```typescript
    /**
     * AUTH-10 / D-06: returns true if the email's domain is on the actively-maintained
     * disposable-email blocklist (disposable-email-domains-js v1.20.x, monthly updates).
     * Used as a Zod refine in the registration server action (plan 1.07).
     * Failure mode: malformed input → false (graceful) so Zod's email() refinement still fires.
     */
    export function isDisposable(email: string): boolean {
      if (!email) return false;
      const trimmed = email.trim().toLowerCase();
      const at = trimmed.lastIndexOf('@');
      if (at <= 0 || at === trimmed.length - 1) return false;
      try {
        return isDisposableEmail(trimmed);
      } catch {
        return false;
      }
    }
    ```

    ```typescript
    /**
     * AUTH-08 / D-05: server-side Cloudflare Turnstile siteverify.
     * RESEARCH § Anti-abuse — fail-closed on any failure mode (network, parse, false).
     */
    const SITEVERIFY = 'https://challenges.cloudflare.com/turnstile/v0/siteverify';

    export interface TurnstileResult {
      ok: boolean;
      errorCodes?: string[];
    }

    export async function verifyTurnstile(token: string, remoteIp?: string): Promise<TurnstileResult> {
      if (!token) return { ok: false, errorCodes: ['missing-input-response'] };
      const secret = process.env.TURNSTILE_SECRET_KEY;
      if (!secret) return { ok: false, errorCodes: ['missing-secret'] };

      const body = new URLSearchParams({ secret, response: token });
      if (remoteIp) body.set('remoteip', remoteIp);

      try {
        const res = await fetch(SITEVERIFY, {
          method: 'POST',
          headers: { 'content-type': 'application/x-www-form-urlencoded' },
          body: body.toString(),
        });
        if (!res.ok) return { ok: false, errorCodes: [`http-${res.status}`] };
        const json = (await res.json()) as { success: boolean; 'error-codes'?: string[] };
        return json.success
          ? { ok: true }
          : { ok: false, errorCodes: json['error-codes'] ?? ['unknown'] };
      } catch {
        return { ok: false, errorCodes: ['fetch-failed'] };
      }
    }
    ```

    **Step 3.** Create `src/lib/ip.ts` (D-22 — IP extraction from CF headers):
    ```typescript
    /**
     * D-22: All traffic passes through Cloudflare. The real client IP is in
     * `cf-connecting-ip`. Fall back to first hop of `x-forwarded-for` for dev.
     * Returns null if no header is present.
     */
    export function getClientIp(headers: Headers): string | null {
      const cf = headers.get('cf-connecting-ip');
      if (cf) return cf.trim();
      const xff = headers.get('x-forwarded-for');
      if (xff) return xff.split(',')[0]?.trim() ?? null;
      return null;
    }

    /**
     * Convert IPv4 address to /24 subnet string (e.g. "1.2.3.42" → "1.2.3.0/24").
     * Used by the registration-subnet rate limiter (D-07 Pitfall 8).
     * IPv6 returns the input unchanged (subnet aggregation is more complex; defer to v2).
     */
    export function getSubnet(ip: string): string {
      if (!/^\d{1,3}(\.\d{1,3}){3}$/.test(ip)) return ip;
      const parts = ip.split('.');
      return `${parts[0]}.${parts[1]}.${parts[2]}.0/24`;
    }
    ```
