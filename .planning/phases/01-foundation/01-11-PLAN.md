    ```typescript
    import pino, { type LoggerOptions, type Logger } from 'pino';

    /**
     * D-21 + PATTERNS § Logger Interface.
     * Allowed: userId, requestId, route, event, durationMs, statusCode, errorCode.
     * Forbidden: email, name, full_name, ip, x-forwarded-for, cf-connecting-ip.
     */
    const REDACT = [
      'email',
      'password',
      'ip',
      'x-forwarded-for',
      'cf-connecting-ip',
      'name',
      'full_name',
    ];

    function buildOptions(): LoggerOptions {
      const isProd = process.env.NODE_ENV === 'production';
      const base: LoggerOptions = {
        level: process.env.LOG_LEVEL ?? 'info',
        redact: { paths: REDACT, censor: '[Redacted]' },
        formatters: { level: (label) => ({ level: label }) },
        timestamp: pino.stdTimeFunctions.isoTime,
      };
      if (isProd && process.env.BETTERSTACK_SOURCE_TOKEN) {
        base.transport = {
          target: '@logtail/pino',
          options: { sourceToken: process.env.BETTERSTACK_SOURCE_TOKEN },
        };
      } else if (!isProd) {
        base.transport = { target: 'pino-pretty', options: { translateTime: 'SYS:standard' } };
      }
      return base;
    }

    export const logger: Logger = pino(buildOptions());
    ```
