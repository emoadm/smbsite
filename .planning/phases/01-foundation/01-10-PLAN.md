    ```typescript
    import { Queue } from 'bullmq';
    import IORedis from 'ioredis';

    /**
     * NOTIF-08 / D-19. Queue boundary: server actions enqueue here in <50ms;
     * worker (src/lib/email/worker.ts) consumes and calls Brevo.
     *
     * RESEARCH Pitfall E: Upstash + BullMQ requires these EXACT settings.
     */
    export type EmailJobKind = 'register-otp' | 'login-otp' | 'welcome';

    export interface EmailJobPayload {
      to: string;
      kind: EmailJobKind;
      otpCode?: string;
      expiresAt?: Date;
      fullName?: string;
    }

    export const EMAIL_QUEUE_NAME = 'email-queue';

    let _connection: IORedis | null = null;
    function getConnection(): IORedis {
      if (_connection) return _connection;
      const url = process.env.UPSTASH_REDIS_URL;
      if (!url) throw new Error('UPSTASH_REDIS_URL not configured');
      _connection = new IORedis(url, {
        maxRetriesPerRequest: null, // Pitfall E
        enableReadyCheck: false,    // Pitfall E
        tls: url.startsWith('rediss://') ? {} : undefined,
      });
      return _connection;
    }

    let _queue: Queue<EmailJobPayload> | null = null;
    function getQueue(): Queue<EmailJobPayload> {
      if (_queue) return _queue;
      _queue = new Queue<EmailJobPayload>(EMAIL_QUEUE_NAME, { connection: getConnection() });
      return _queue;
    }

    /**
     * Add an email job. Returns immediately; worker delivers asynchronously.
     * Tests can spy on this function. Production callers must always await.
     *
     * H-3 / plan 1.09 hook: when TEST_OTP_SINK is set (E2E only), the payload is appended to
     * the named JSON file. This lets plan 1.09's `tests/e2e/login.spec.ts` (`AUTH-05 + AUTH-07`)
     * read the plaintext OTP and submit it through the UI without depending on a live Brevo send.
     */
    export async function addEmailJob(payload: EmailJobPayload): Promise<void> {
      const sinkPath = process.env.TEST_OTP_SINK;
      if (sinkPath && process.env.NODE_ENV !== 'production') {
        try {
          const fs = await import('node:fs/promises');
          const dir = (await import('node:path')).dirname(sinkPath);
          await fs.mkdir(dir, { recursive: true });
          const existing = await fs.readFile(sinkPath, 'utf8').catch(() => '[]');
          let arr: EmailJobPayload[] = [];
          try { arr = JSON.parse(existing); if (!Array.isArray(arr)) arr = []; } catch { arr = []; }
          arr.push(payload);
          await fs.writeFile(sinkPath, JSON.stringify(arr));
        } catch { /* sink errors must not break the action */ }
      }
      // Test/dev shortcut: when UPSTASH_REDIS_URL is unset (vitest), no-op so unit tests work.
      if (!process.env.UPSTASH_REDIS_URL) {
        if (process.env.NODE_ENV === 'production') {
          throw new Error('UPSTASH_REDIS_URL must be set in production');
        }
        return;
      }
      await getQueue().add(payload.kind, payload, {
        attempts: 5,
        backoff: { type: 'exponential', delay: 5_000 },
        removeOnComplete: { age: 24 * 3600, count: 1000 },
        removeOnFail: { age: 7 * 24 * 3600 },
      });
    }

    /** For graceful shutdown. */
    export async function closeQueue(): Promise<void> {
      if (_queue) await _queue.close();
      if (_connection) await _connection.quit();
      _queue = null;
      _connection = null;
    }
    ```

    **Step 3.** `src/lib/email/brevo.ts` — minimal Brevo REST client (transactional/email send only):
    ```typescript
    /**
     * Brevo transactional send. Single endpoint suffices for Phase 1.
     * Pitfall I: Brevo encodes Cyrillic subject lines automatically (UTF-8 base64).
     * Pitfall J: include both `htmlContent` AND `textContent` for Abv.bg deliverability.
     */
    interface BrevoSendArgs {
      to: { email: string; name?: string };
      subject: string;
      htmlContent: string;
      textContent: string;
      from?: { email: string; name?: string };
    }

    export async function sendBrevoEmail(args: BrevoSendArgs): Promise<{ messageId: string }> {
      const apiKey = process.env.BREVO_API_KEY;
      if (!apiKey) throw new Error('BREVO_API_KEY not configured');
      const fromEmail = args.from?.email ?? process.env.EMAIL_FROM_TRANSACTIONAL ?? 'no-reply@auth.example.invalid';

      const res = await fetch('https://api.brevo.com/v3/smtp/email', {
        method: 'POST',
        headers: {
          'api-key': apiKey,
          'content-type': 'application/json',
          accept: 'application/json',
        },
        body: JSON.stringify({
          to: [args.to],
          sender: { email: fromEmail, name: args.from?.name ?? 'Синя България' },
          subject: args.subject,
          htmlContent: args.htmlContent,
          textContent: args.textContent, // Pitfall J — required for Abv.bg
        }),
      });

      if (!res.ok) {
        const body = await res.text();
        throw new Error(`Brevo send failed: ${res.status} ${body.slice(0, 200)}`);
      }
      const json = (await res.json()) as { messageId: string };
      return { messageId: json.messageId };
    }
    ```

    **Step 4.** `src/lib/email/worker.ts` — BullMQ worker:
    ```typescript
    import { Worker, type Job } from 'bullmq';
    import IORedis from 'ioredis';
    import { render } from '@react-email/render';
    import { getTranslations } from 'next-intl/server';
    import { OtpEmail } from './templates/OtpEmail';
    import { LoginOtpEmail } from './templates/LoginOtpEmail';
    import { WelcomeEmail } from './templates/WelcomeEmail';
    import { sendBrevoEmail } from './brevo';
    import { EMAIL_QUEUE_NAME, type EmailJobPayload } from './queue';

    function workerConnection(): IORedis {
      const url = process.env.UPSTASH_REDIS_URL!;
      return new IORedis(url, {
        maxRetriesPerRequest: null,
        enableReadyCheck: false,
        tls: url.startsWith('rediss://') ? {} : undefined,
      });
    }

    /**
     * B-1 + H-5 fix: worker resolves Bulgarian copy via getTranslations({ locale: 'bg', namespace: 'email.<kind>' }).
     * Templates receive a pre-built `t` object so the .tsx files contain ZERO Cyrillic literals
     * (D-27 / PUB-05).
     */
    type EmailT = (key: string, vars?: Record<string, string | number>) => string;

    async function loadT(namespace: string): Promise<EmailT> {
      const raw = await getTranslations({ locale: 'bg', namespace });
      return (key: string, vars?: Record<string, string | number>) => raw(key, vars);
    }

    async function processor(job: Job<EmailJobPayload>): Promise<{ messageId: string }> {
      const { to, kind, otpCode, fullName } = job.data;
      let subject = '';
      let html = '';
      let text = '';

      switch (kind) {
        case 'register-otp': {
          const t = await loadT('email.registerOtp');
          subject = t('subject');
          const props = { t, code: otpCode!, fullName: fullName ?? '', validityHours: 48 };
          html = await render(<OtpEmail {...props} />);
          text = await render(<OtpEmail {...props} />, { plainText: true });
          break;
        }
        case 'login-otp': {
          const t = await loadT('email.loginOtp');
          subject = t('subject');
          const props = { t, code: otpCode!, validityMinutes: 10 };
          html = await render(<LoginOtpEmail {...props} />);
          text = await render(<LoginOtpEmail {...props} />, { plainText: true });
          break;
        }
        case 'welcome': {
          const t = await loadT('email.welcome');
          subject = t('subject');
          const props = { t, fullName: fullName ?? '' };
          html = await render(<WelcomeEmail {...props} />);
          text = await render(<WelcomeEmail {...props} />, { plainText: true });
          break;
        }
      }

      const tFrom = await loadT('email.from');
      return sendBrevoEmail({
        to: { email: to, name: fullName },
        subject,
        htmlContent: html,
        textContent: text,
        from: { email: process.env.EMAIL_FROM_TRANSACTIONAL ?? 'no-reply@auth.example.invalid', name: tFrom('name') },
      });
    }

    export function startWorker() {
      return new Worker<EmailJobPayload>(EMAIL_QUEUE_NAME, processor as any, {
        connection: workerConnection(),
        concurrency: 3,
      });
    }
    ```
