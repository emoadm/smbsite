       ```typescript
       'use server';

       import { headers } from 'next/headers';
       import { eq } from 'drizzle-orm';
       import { auth } from '@/lib/auth';
       import { db } from '@/db';
       import { submissions, users } from '@/db/schema';
       import { proposalSchema, problemReportSchema } from './zod';
       import { verifyTurnstile } from '@/lib/turnstile';
       import { getClientIp } from '@/lib/ip';
       import { checkSubmissionPerUser, checkSubmissionPerIp } from '@/lib/rate-limit';

       export type SubmitResult =
         | { ok: true; nextHref: string }
         | { ok: false; error?: string; fieldErrors?: Record<string, string[]> };

       async function loadSessionUser() {
         const session = await auth();
         if (!session?.user) return null;
         return session.user as { id: string; emailVerified?: Date | null };
       }

       async function checkActiveStatus(userId: string): Promise<'active' | 'suspended' | 'deleted' | null> {
         const [u] = await db
           .select({ status: users.status })
           .from(users)
           .where(eq(users.id, userId))
           .limit(1);
         return (u?.status as 'active' | 'suspended' | 'deleted' | undefined) ?? null;
       }

       export async function submitProposal(
         _prevState: SubmitResult | null,
         formData: FormData,
       ): Promise<SubmitResult> {
         // 1. Auth + email-verified
         const sessionUser = await loadSessionUser();
         if (!sessionUser) return { ok: false, error: 'submission.gate.unverified' };
         if (!sessionUser.emailVerified) return { ok: false, error: 'submission.gate.unverified' };

         // 2. Suspension gate (defense in depth alongside member/layout.tsx)
         const status = await checkActiveStatus(sessionUser.id);
         if (status === 'suspended') return { ok: false, error: 'submission.gate.suspended' };

         // 3. Rate limit (per-user + per-IP)
         const h = await headers();
         const ip = getClientIp(h);
         const userR = await checkSubmissionPerUser(sessionUser.id);
         if (!userR.success) return { ok: false, error: 'submission.error.rateLimit' };
         if (ip) {
           const ipR = await checkSubmissionPerIp(ip);
           if (!ipR.success) return { ok: false, error: 'submission.error.rateLimit' };
         }

         // 4. Zod parse
         const raw = Object.fromEntries(formData.entries());
         const parsed = proposalSchema.safeParse(raw);
         if (!parsed.success) {
           return { ok: false, fieldErrors: parsed.error.flatten().fieldErrors };
         }

         // 5. Turnstile
         const tr = await verifyTurnstile(parsed.data['cf-turnstile-response'], ip ?? undefined);
         if (!tr.ok) return { ok: false, error: 'submission.error.captchaFailed' };

         // 6. INSERT
         await db.insert(submissions).values({
           submitter_id: sessionUser.id,
           kind: 'proposal',
           status: 'pending',
           title: parsed.data.title,
           body: parsed.data.body,
           topic: parsed.data.topic,
         });

         return { ok: true, nextHref: '/member/predlozheniya' };
       }

       export async function submitProblemReport(
         _prevState: SubmitResult | null,
         formData: FormData,
       ): Promise<SubmitResult> {
         const sessionUser = await loadSessionUser();
         if (!sessionUser) return { ok: false, error: 'submission.gate.unverified' };
         if (!sessionUser.emailVerified) return { ok: false, error: 'submission.gate.unverified' };

         const status = await checkActiveStatus(sessionUser.id);
         if (status === 'suspended') return { ok: false, error: 'submission.gate.suspended' };

         const h = await headers();
         const ip = getClientIp(h);
         const userR = await checkSubmissionPerUser(sessionUser.id);
         if (!userR.success) return { ok: false, error: 'submission.error.rateLimit' };
         if (ip) {
           const ipR = await checkSubmissionPerIp(ip);
           if (!ipR.success) return { ok: false, error: 'submission.error.rateLimit' };
         }

         const raw = Object.fromEntries(formData.entries());
         const parsed = problemReportSchema.safeParse(raw);
         if (!parsed.success) {
           return { ok: false, fieldErrors: parsed.error.flatten().fieldErrors };
         }

         const tr = await verifyTurnstile(parsed.data['cf-turnstile-response'], ip ?? undefined);
         if (!tr.ok) return { ok: false, error: 'submission.error.captchaFailed' };

         await db.insert(submissions).values({
           submitter_id: sessionUser.id,
           kind: 'problem',
           status: 'pending',
           body: parsed.data.body,
           topic: parsed.data.topic,
           level: parsed.data.level,
           oblast: parsed.data.level === 'local' ? (parsed.data.oblast ?? null) : null,
         });

         return { ok: true, nextHref: '/member/signali' };
       }
       ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Anonymous browser → submitProposal/submitProblemReport Server Actions | Server Actions are reachable directly without going through the Next.js form (RPC-style); auth + rate-limit + Turnstile + Zod must all run server-side regardless of UI state |
| Member browser → request IP → MaxMind GeoIP | IP exists only on the server; never echoed to the client; the client receives only the derived ISO oblast code (D-19 / GDPR-09 inheritance) |
| Suspended account → submission Server Action | Layout-level gate is bypassable via direct RPC call; assertNotSuspended runs INSIDE every Server Action |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-03-01 | Spoofing | Sockpuppet submission spam from one user | mitigate | Upstash sliding-window 5/24h per user_id (`submission-user`) + 10/24h per IP (`submission-ip`); Turnstile on every submit. ASVS V11.1. |
| T-04-03-02 | Tampering | Form submitted with invalid kind/status (e.g., `status: 'approved'` injected by attacker) | mitigate | Server Actions never accept `kind` or `status` from FormData — both are hardcoded ('proposal'|'problem' kind; 'pending' status). Zod schema does not include these fields. |
| T-04-03-03 | Information Disclosure | Raw IP leaked to client via prop drilling | mitigate | Page only passes `defaultOblastCode: string | null` (the derived ISO code). No IP-related field. |
| T-04-03-04 | Information Disclosure | Server Action error messages leak stack traces or DB internals | mitigate | All error paths return typed `{ ok: false, error: 'submission.error.<key>' }`; no `throw` reaches the client; Turnstile/Zod/ratelimit fall through to typed errors. ASVS V7.1. |
| T-04-03-05 | Elevation of Privilege | Suspended account submits via direct Server Action call (skipping layout) | mitigate | `checkActiveStatus(userId)` runs at step 2 of every Server Action — defense-in-depth alongside layout-level gate (Plan 04-07). ASVS V4.1. |
| T-04-03-06 | Tampering | Member submits problem report with level='local' but no oblast (Zod path is the only enforcement) | mitigate | proposalSchema/problemReportSchema use `superRefine` to require oblast when level='local' AND a regex `^BG-(0[1-9]|1[0-9]|2[0-8])$` to reject arbitrary strings (defense-in-depth on top of free-text DB column). |
| T-04-03-07 | Denial of Service | Burst submissions via parallel tabs evade per-user rate-limit (Upstash atomicity) | accept | Upstash Ratelimit uses Redis SET NX semantics atomically; concurrent calls compete for the same window slot. The 10/IP-24h cap is the safety net. Phase 6 OPS-05 load test will reveal if 5/user/24h is too generous. |
| T-04-03-08 | Information Disclosure | Bulgarian Cyrillic URL `/member/предложи` rejected by CDN/origin path encoding | accept | Latin transliteration on disk is the project standard; nav labels are Bulgarian, hrefs are Latin. Cyrillic URL routing can be added as a follow-up middleware rewrite (out of scope here). |
