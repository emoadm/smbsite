<scope_note>
DSA Art.16 minimum compliance is INTENTIONALLY narrow:
- Logged-in members only (anonymous DSA reports omitted in v1; Art.16 allows but does not require accepting anonymous reports for hosting services that don't otherwise serve anonymous users)
- Reportable surface: only public ProposalCards on /predlozheniya. The /problemi heat-map shows aggregates without identifiable content, so per-DSA "content" reporting is not applicable.
- Acknowledgement: on-screen toast + email (the email reuses the existing Plan 04-07 worker pattern with a new job kind, OR is deferred to a follow-up — Phase 4 ships the on-screen acknowledgement and writes the queue row; the email is a follow-up note in SUMMARY.md if not landed)
- Decision notification: when an editor approves a DSA report (treating "approve" as "we agree the content was problematic and have hidden it") OR rejects it ("we disagree"), the standard submission-status email fires. The reporter is the submitter of the DSA report, so the existing Plan 04-07 status-email path notifies them. This is a deliberate reuse of infrastructure.

Editors handling DSA reports use the SAME Plan 04-06 ModerationQueueView "DSA сигнали" tab — no new admin UI. The reviewer reads the DSA report's body (reason) and topic (category), then approves (= "we agree, content was problematic — also hide the original via super_editor_override" — Phase 4 ships this as a manual two-step) or rejects (= "we reviewed, no action needed").

If this scope feels insufficient under DSA enforcement scrutiny, escalate as a follow-up plan after Phase 4 ships. CONTEXT.md explicitly defers "DSA Art.16 reporting mechanism scope decision" to the planner.
</scope_note>

    2. Create `src/lib/submissions/dsa-actions.ts`:
       ```typescript
       'use server';

       import { headers } from 'next/headers';
       import { eq } from 'drizzle-orm';
       import { auth } from '@/lib/auth';
       import { db } from '@/db';
       import { submissions, users } from '@/db/schema';
       import { dsaReportSchema } from './zod';
       import { verifyTurnstile } from '@/lib/turnstile';
       import { getClientIp } from '@/lib/ip';
       import { checkSubmissionPerUser, checkSubmissionPerIp } from '@/lib/rate-limit';

       export type DsaSubmitResult =
         | { ok: true }
         | { ok: false; error?: string; fieldErrors?: Record<string, string[]> };

       export async function submitDsaReport(
         _prevState: DsaSubmitResult | null,
         formData: FormData,
       ): Promise<DsaSubmitResult> {
         const session = await auth();
         if (!session?.user) return { ok: false, error: 'submission.gate.unverified' };
         const sessionUser = session.user as { id: string; emailVerified?: Date | null };
         if (!sessionUser.emailVerified) return { ok: false, error: 'submission.gate.unverified' };

         // Suspension gate
         const [u] = await db.select({ status: users.status }).from(users).where(eq(users.id, sessionUser.id)).limit(1);
         if (u?.status === 'suspended') return { ok: false, error: 'submission.gate.suspended' };

         // Rate limit (reuse PROP/PROB limiters — DSA reports count against the same budget)
         const h = await headers();
         const ip = getClientIp(h);
         const userR = await checkSubmissionPerUser(sessionUser.id);
         if (!userR.success) return { ok: false, error: 'submission.error.rateLimit' };
         if (ip) {
           const ipR = await checkSubmissionPerIp(ip);
           if (!ipR.success) return { ok: false, error: 'submission.error.rateLimit' };
         }

         const raw = Object.fromEntries(formData.entries());
         const parsed = dsaReportSchema.safeParse(raw);
         if (!parsed.success) {
           return { ok: false, fieldErrors: parsed.error.flatten().fieldErrors };
         }

         const tr = await verifyTurnstile(parsed.data['cf-turnstile-response'], ip ?? undefined);
         if (!tr.ok) return { ok: false, error: 'submission.error.captchaFailed' };

         await db.insert(submissions).values({
           submitter_id: sessionUser.id,
           kind: 'dsa_report',
           status: 'pending',
           body: parsed.data.reason,
           topic: parsed.data.category, // category serves as topic; editor sees it in queue
           target_submission_id: parsed.data.targetSubmissionId,
         });

         return { ok: true };
       }
       ```

       import { useActionState, useEffect, useState } from 'react';
       import { useTranslations } from 'next-intl';
       import { Dialog, DialogContent, DialogHeader, DialogTitle, DialogDescription } from '@/components/ui/dialog';

       interface Props {
         submissionId: string;
         open: boolean;
         onOpenChange: (open: boolean) => void;
       }

       const CATEGORY_KEYS = ['illegal', 'harassment', 'misinformation', 'spam', 'other'] as const;
       const initialState: DsaSubmitResult = { ok: false };

       export function ReportContentDialog({ submissionId, open, onOpenChange }: Props) {
         const t = useTranslations('dsa.report');
         const tCommon = useTranslations('submission.error');
         const [state, formAction, pending] = useActionState(submitDsaReport, initialState);
         const [turnstileStatus, setTurnstileStatus] = useState<'loading' | 'ready' | 'error'>('loading');
         const [category, setCategory] = useState<string>('');

         useEffect(() => {
           if (!open) {
             // Reset on close
             setCategory('');
           }
         }, [open]);

         return (
           <Dialog open={open} onOpenChange={onOpenChange}>
             <DialogContent className="max-w-md">
               <DialogHeader>
                 <DialogTitle>{t('heading')}</DialogTitle>
                 <DialogDescription>{t('body')}</DialogDescription>
               </DialogHeader>

               {state.ok ? (
                 <div className="space-y-4 py-4">
                   <h3 className="font-display text-lg font-extrabold">{t('successHeading')}</h3>
                   <p className="text-sm text-muted-foreground">{t('successBody')}</p>
                   <Button variant="outline" onClick={() => onOpenChange(false)}>Затвори</Button>
                 </div>
               ) : (

                   <input type="hidden" name="targetSubmissionId" value={submissionId} />

                     <div role="alert" className="text-destructive text-sm">
                       {state.error.startsWith('submission.error') ? tCommon(state.error.replace('submission.error.', '') as 'rateLimit' | 'captchaFailed' | 'validation') : state.error}
                     </div>
                   )}

                   <div className="flex flex-col gap-2">
                     <Label htmlFor="category">{t('categoryLabel')}</Label>
                     <Select name="category" value={category} onValueChange={setCategory}>
                       <SelectTrigger id="category"><SelectValue /></SelectTrigger>
                       <SelectContent>
                         {CATEGORY_KEYS.map((key) => (
                           <SelectItem key={key} value={key}>
                             {t(`categories.${key}` as 'categories.illegal')}
                           </SelectItem>
                         ))}
                       </SelectContent>
                     </Select>
                     {state.fieldErrors?.category && (
                       <p className="text-destructive text-sm">{state.fieldErrors.category[0]}</p>
                     )}
                   </div>

                   <div className="flex flex-col gap-2">
                     <Label htmlFor="reason">{t('reasonLabel')}</Label>
                     <Textarea id="reason" name="reason" rows={4} placeholder={t('reasonPlaceholder')} required minLength={20} />

                   </Button>
                 </form>
