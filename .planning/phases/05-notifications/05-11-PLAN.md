  // Adapt to whatever editor-login fixture exists; Phase 02.1 uses a similar

  await page.getByLabel(/email/i).fill(process.env.E2E_EDITOR_EMAIL ?? 'editor@chastnik.eu');

  await page.getByLabel(/тема на бюлетина/i).click();
  await page.getByText('Общи обявявания').click();
  // Lexical RTE — focus + type
  const editor = page.locator('[contenteditable="true"]').first();
  await editor.click();

  // 3. Save draft (Payload's native save)
  await page.getByRole('button', { name: /save|запази/i }).first().click();
  await expect(page.getByText(/saved|записано/i).first()).toBeVisible({ timeout: 5000 });

  // 4. Verify Send Blast is disabled (no test sent yet)
  const sendBlastBtn = page.getByRole('button', { name: /Изпрати рекламата/ });
  await expect(sendBlastBtn).toBeDisabled();

  // 5. Click Send Test to me
  await page.getByRole('button', { name: /Изпрати тестово писмо до мен/ }).click();
  // Wait for Sonner toast confirming success
  await expect(page.getByText(/Тестовото писмо е изпратено/)).toBeVisible({ timeout: 10000 });

  // 6. Reload to refresh lastTestSentAt
  await page.reload();
  // After reload, Send Blast must be enabled
  await expect(page.getByRole('button', { name: /Изпрати рекламата/ })).toBeEnabled({ timeout: 5000 });

  // 7. Click Send Blast → expect post-send Dialog
  await page.getByRole('button', { name: /Изпрати рекламата/ }).click();
