    **File 1 — `src/lib/cookie-consent/persist.ts`**:

    ```typescript
    import { randomUUID, createHash } from 'node:crypto';
    import { db } from '@/db';
    import { cookie_consents } from '@/db/schema';
    import { lookupRegion } from '@/lib/attribution/geoip'; // confirm export name

    export interface PersistCookieConsentInput {
      attrSidCookie: string | undefined;   // value from attr_sid cookie
      ip: string | null;                   // raw IP from x-forwarded-for / x-real-ip
      choices: Record<string, unknown>;    // CookieYes payload
      userId?: string | null;              // optional — set if session.user.id present
    }

    /**
     * Phase 6 D-09 — Persist anonymous cookie-consent event.
     *
     * - anon_id = attr_sid cookie value (if valid UUID) OR fresh randomUUID().
     * - ip_hash = sha256(ip + daily salt). Raw IP NEVER stored.
     * - region = GeoIP lookup result.
     * - user_id = session.user.id if present (post-OTP linkage path mirrors Phase 02.1 D-07).
     */
    export async function persistCookieConsent(input: PersistCookieConsentInput): Promise<{ anonId: string }> {
      const anonId = isUuid(input.attrSidCookie) ? input.attrSidCookie! : randomUUID();

      const today = new Date().toISOString().slice(0, 10); // daily-rotating salt
      const ipHash = input.ip
        ? createHash('sha256').update(`${input.ip}|${today}`).digest('hex')
        : null;

      const region = input.ip ? await lookupRegion(input.ip).catch(() => null) : null;

      await db.insert(cookie_consents).values({
        anon_id: anonId,
        user_id: input.userId ?? null,
        choices: input.choices,
        ip_hash: ipHash,
        region,
      });

      return { anonId };
    }

    function isUuid(s: string | undefined): s is string {
      return typeof s === 'string' && /^[0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}$/i.test(s);
    }
    ```

    **File 2 — `src/app/api/cookie-consent/route.ts`**:

    ```typescript
    import { NextRequest, NextResponse } from 'next/server';
    import { Ratelimit } from '@upstash/ratelimit';
    import { Redis } from '@upstash/redis';
    import { auth } from '@/lib/auth';
    import { persistCookieConsent } from '@/lib/cookie-consent/persist';

    export const runtime = 'nodejs';

    const limiter = new Ratelimit({
      redis: Redis.fromEnv(),
      limiter: Ratelimit.fixedWindow(1, '1 m'),
      prefix: 'rl:cookie-consent',
    });

    export async function POST(req: NextRequest) {
      // Extract IP (Fly / Cloudflare forward-chain)
      const ip = req.headers.get('x-forwarded-for')?.split(',')[0]?.trim()
        ?? req.headers.get('x-real-ip')
        ?? null;

      // Rate-limit per IP
      if (ip) {
        const rl = await limiter.limit(ip);
        if (!rl.success) {
          return new NextResponse(null, { status: 429 });
        }
      }

      // Parse body
      let body: { choices?: Record<string, unknown> };
      try {
        body = await req.json();
      } catch {
        return NextResponse.json({ error: 'invalid_json' }, { status: 400 });
      }
      const choices = body.choices;
      if (typeof choices !== 'object' || choices === null || Array.isArray(choices)) {
        return NextResponse.json({ error: 'invalid_choices' }, { status: 400 });
      }

      // Read attr_sid cookie
      const attrSidCookie = req.cookies.get('attr_sid')?.value;

      // Optional session for user_id linkage
      const session = await auth().catch(() => null);
      const userId = session?.user?.id ?? null;

      await persistCookieConsent({ attrSidCookie, ip, choices, userId });

      return new NextResponse(null, { status: 204 });
    }
    ```
