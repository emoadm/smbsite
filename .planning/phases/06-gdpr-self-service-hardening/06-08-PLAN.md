    import Link from 'next/link';

    import { Alert, AlertDescription } from '@/components/ui/alert';

    import { auth } from '@/lib/auth';
    import { db } from '@/db';
    import { users } from '@/db/schema';
    import { eq } from 'drizzle-orm';
    import { DataExportRequestButton } from './DataExportRequestButton';

    function formatBulgarianDate(d: Date): string {
      return d.toLocaleDateString('bg-BG', { day: '2-digit', month: '2-digit', year: 'numeric' });
    }

    export async function DataRightsCard() {
      const t = await getTranslations('member.preferences.dataRights');
      const session = await auth();
      const userId = session?.user?.id;

      let exportRequestedAt: Date | null = null;
      let lastExportExpiredAt: Date | null = null;

      if (userId) {
        const [row] = await db
          .select({
            export_requested_at: users.export_requested_at,
            last_export_expired_at: users.last_export_expired_at,
          })
          .from(users)
          .where(eq(users.id, userId))
          .limit(1);
        if (row) {
          exportRequestedAt = row.export_requested_at;
          lastExportExpiredAt = row.last_export_expired_at;
        }
      }

      const showExpiredAlert =
        lastExportExpiredAt !== null &&
        exportRequestedAt !== null &&
        lastExportExpiredAt > exportRequestedAt;

      return (
        <Card>
          <CardHeader>
            <CardTitle>{t('title')}</CardTitle>
            <CardDescription>{t('description')}</CardDescription>
          </CardHeader>
          <CardContent className="space-y-6">
            <p className="text-base leading-normal">
              {t.rich('explainer', {
                privacyLink: (chunks) => (
                  <Link href="/privacy" className="text-primary underline-offset-4 hover:underline">
                    {chunks}
                  </Link>
                ),
              })}

                  {t('export.lastRequest', { date: formatBulgarianDate(exportRequestedAt) })}

                <Alert>
                  <AlertDescription>{t('export.expiredAlert')}</AlertDescription>

    **File 2 — `src/components/preferences/DataExportRequestButton.tsx`** (client component):

    ```tsx
    'use client';

    import { useState, useTransition } from 'react';
    import { useTranslations } from 'next-intl';
    import { Loader2, CheckCircle2 } from 'lucide-react';
    import { toast } from 'sonner';
    import { Alert, AlertDescription } from '@/components/ui/alert';
    import { Button } from '@/components/ui/button';
    import { requestExport } from '@/lib/data-rights/actions';

    type AlertState =
      | { kind: 'none' }
      | { kind: 'error' }
      | { kind: 'rateLimit'; minutes: number };

    const COOLDOWN_MS = 5 * 60 * 1000; // 5 minutes per UI-SPEC §S5

    export function DataExportRequestButton() {
      const t = useTranslations('member.preferences.dataRights.export');
      const [isPending, startTransition] = useTransition();
      const [cooldownUntil, setCooldownUntil] = useState<number | null>(null);
      const [alertState, setAlertState] = useState<AlertState>({ kind: 'none' });

      const isCooldown = cooldownUntil !== null && Date.now() < cooldownUntil;
      const disabled = isPending || isCooldown;

      function handleClick() {
        setAlertState({ kind: 'none' });
        startTransition(async () => {
          const result = await requestExport({});
          if (result.ok) {
            toast.success(t('successToast'));
            setCooldownUntil(Date.now() + COOLDOWN_MS);
            // Re-enable after cooldown
            setTimeout(() => setCooldownUntil(null), COOLDOWN_MS + 1000);
          } else if (result.code === 'rate_limited') {
            const minutes = Math.ceil((result.retryAfter ?? 60) / 60);
            setAlertState({ kind: 'rateLimit', minutes });
          } else {
            setAlertState({ kind: 'error' });
          }
        });
      }

      const showSuccessLabel = isCooldown && !isPending;

      return (
        <div className="space-y-2">
          <Button
            type="button"
            variant="default"
            disabled={disabled}
            aria-busy={isPending}
            onClick={handleClick}
          >
            {isPending && <Loader2 className="size-4 animate-spin mr-2" aria-hidden="true" />}
            {showSuccessLabel && <CheckCircle2 className="size-4 mr-2" aria-hidden="true" />}
            {showSuccessLabel ? t('ctaSent') : t('cta')}
          </Button>

          {alertState.kind === 'error' && (
            <Alert variant="destructive" role="alert">
              <AlertDescription>{t('errorAlert')}</AlertDescription>
            </Alert>
          )}
          {alertState.kind === 'rateLimit' && (
            <Alert variant="destructive" role="alert">
              <AlertDescription>{t('rateLimitAlert', { minutes: String(alertState.minutes) })}</AlertDescription>
            </Alert>
          )}
        </div>
      );
    }
    ```
