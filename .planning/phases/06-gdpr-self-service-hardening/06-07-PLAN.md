- `requestDeletion` flips `users.status` to 'pending_deletion' + sets `deletion_requested_at = now()` ONLY if current `status === 'active'`. If user is already in pending_deletion (re-request without cancel), the action returns idempotent success.

- DO NOT call `signOut()` directly in `requestDeletion` Server Action — that's the responsibility of the calling form (UI redirects after the action returns; signOut runs on the page transition). The action sets up the DB state; the UI handles the auth lifecycle.

    /**
     * Phase 6 D-04 — typed-text confirmation target string.
     *
     * Module-level constant (NOT an i18n string). Per RESEARCH.md Pitfall G:
     * - Validation logic compares input to this literal verbatim.
     * - The user-facing prompt ("Въведете точно ИЗТРИЙТЕ") lives in messages/bg.json
     *   under member.deleteAccount.typedConfirmTarget.
     * - i18n linter doesn't apply to validation primitives.
     *
     * Cyrillic uppercase, 8 characters, no whitespace, no punctuation.
     * Case-sensitive + script-sensitive (Phase 1 D-27 Bulgarian formal-respectful contract).
     */
    export const TYPED_DELETE_CONFIRM = 'ИЗТРИЙТЕ' as const;

    /**
     * requestExport — no client-side payload beyond session check.
     * Server reads userId from session, not from client. Schema validates the
     * empty-object envelope just to keep the action signature uniform.
     */
    export const requestExportSchema = z.object({}).strict();

    /**
     * requestDeletion — typed-text confirmation only.
     * trim() applied before exact-match per UI-SPEC §S2 (paste-with-trailing-space tolerance).
     */
    export const requestDeletionSchema = z.object({
      typedConfirm: z
        .string()
        .transform((s) => s.trim())
        .refine((s) => s === TYPED_DELETE_CONFIRM, {
          message: 'Текстът за потвърждение не съвпада.',
        }),
    });

    /**
     * requestCancelDeletionOtp — no payload (user is on /account-pending-deletion;
     * server reads their email from the session).
     */
    export const requestCancelDeletionOtpSchema = z.object({}).strict();

    /**
     * verifyCancelDeletionOtp — 6-digit OTP code.
     */
    export const cancelDeletionOtpSchema = z.object({
      code: z
        .string()
        .trim()
        .regex(/^\d{6}$/, 'Кодът трябва да съдържа 6 цифри.'),
    });

    ```typescript
    'use server';

    import { revalidatePath } from 'next/cache';
    import { eq } from 'drizzle-orm';
    import { auth } from '@/lib/auth';
    import { db } from '@/db';
    import { users } from '@/db/schema';
    import { addEmailJob, getQueue } from '@/lib/email/queue';
    import { persistHashedOtp, verifyHashedOtp } from '@/lib/auth';  // verify export names from existing auth.ts
    import {
      requestExportSchema,
      requestDeletionSchema,
      requestCancelDeletionOtpSchema,
      cancelDeletionOtpSchema,
    } from './zod';
    import {
      exportRateLimit,
      deletionRateLimit,
      cancelOtpRateLimit,
      isRateLimitBypassed,
    } from './rate-limits';

    type ActionResult<T = unknown> =
      | { ok: true; data?: T }
      | { ok: false; code: 'unauthorized' | 'rate_limited' | 'invalid_input' | 'wrong_status' | 'server_error'; message?: string; retryAfter?: number };

    async function requireUserId(): Promise<string> {
      const session = await auth();
      const userId = session?.user?.id;
      if (!userId) throw new Error('unauthorized');
      return userId;
    }

    // ============================================================================
    // GDPR-04 — Data export request
    // ============================================================================
    export async function requestExport(_input: unknown = {}): Promise<ActionResult> {
      let userId: string;
      try { userId = await requireUserId(); } catch { return { ok: false, code: 'unauthorized' }; }

      const parsed = requestExportSchema.safeParse(_input);
      if (!parsed.success) return { ok: false, code: 'invalid_input' };

      if (!isRateLimitBypassed()) {
        const rl = await exportRateLimit.limit(userId);
        if (!rl.success) {
          return { ok: false, code: 'rate_limited', retryAfter: Math.ceil((rl.reset - Date.now()) / 1000) };
        }
      }

      // Stamp export_requested_at on users (drives the S5 "Последно искане" line)
      await db
        .update(users)
        .set({ /* export_requested_at: new Date()  — verify whether column exists; add to Plan 06-02 0004 if missing */ })
        .where(eq(users.id, userId));

      await addEmailJob({ to: '', kind: 'data-export', userId });

      revalidatePath('/member/preferences');
      return { ok: true };
    }

    // ============================================================================
    // GDPR-05 — Account deletion request (typed-confirm)
    // ============================================================================
    export async function requestDeletion(input: { typedConfirm: string }): Promise<ActionResult> {
      let userId: string;
      try { userId = await requireUserId(); } catch { return { ok: false, code: 'unauthorized' }; }

      const parsed = requestDeletionSchema.safeParse(input);
      if (!parsed.success) return { ok: false, code: 'invalid_input' };

      if (!isRateLimitBypassed()) {
        const rl = await deletionRateLimit.limit(userId);
        if (!rl.success) {
          return { ok: false, code: 'rate_limited', retryAfter: Math.ceil((rl.reset - Date.now()) / 1000) };
        }
      }

      // Flip status active → pending_deletion only; concurrent run safety.
      const deletionRequestedAt = new Date();
      const updated = await db
        .update(users)
        .set({
          status: 'pending_deletion',
          deletion_requested_at: deletionRequestedAt,
        })
        .where(eq(users.id, userId))  // verify status=active guard via additional where clause
        .returning({ id: users.id, status: users.status });

      if (updated.length === 0) return { ok: false, code: 'wrong_status' };

      // Two BullMQ jobs: immediate email + delayed cascade.
      await addEmailJob({
        to: '',
        kind: 'account-deletion-pending',
        userId,
        deletionRequestedAt,
      });

      // Delayed cascade with jobId so cancelDeletion can remove it.
      const q = getQueue();
      await q.add(
        'account-deletion-cascade',
        { to: '', kind: 'account-deletion-cascade', userId },
        {
          delay: 30 * 86_400_000,            // 30 days
          jobId: `account-deletion-cascade:${userId}`,
        },
      );

      revalidatePath('/member');
      revalidatePath('/member/preferences');
      return { ok: true };
    }

    // ============================================================================
    // GDPR-05 — Cancel-deletion OTP send
    // ============================================================================
    export async function requestCancelDeletionOtp(_input: unknown = {}): Promise<ActionResult> {
      let userId: string;
      try { userId = await requireUserId(); } catch { return { ok: false, code: 'unauthorized' }; }

      const parsed = requestCancelDeletionOtpSchema.safeParse(_input);
      if (!parsed.success) return { ok: false, code: 'invalid_input' };

      if (!isRateLimitBypassed()) {
        const rl = await cancelOtpRateLimit.limit(userId);
        if (!rl.success) {
          return { ok: false, code: 'rate_limited', retryAfter: Math.ceil((rl.reset - Date.now()) / 1000) };
        }
      }

      // Defense-in-depth: verify still in pending_deletion
      const [row] = await db.select({ status: users.status, email: users.email })
        .from(users).where(eq(users.id, userId)).limit(1);
      if (!row || row.status !== 'pending_deletion') {
        return { ok: false, code: 'wrong_status' };
      }
      if (!row.email) return { ok: false, code: 'server_error', message: 'email missing on user row' };

      // Issue OTP — reuses Phase 1 persistHashedOtp helper.
      // verificationTokens.kind = 'cancel_deletion' (new enum value — confirm whether existing kind column accepts free-text or requires migration).
      // If kind column is restricted, fall back to using 'login' as the kind and disambiguate by recent timestamp; document in SUMMARY.
      const otpCode = await persistHashedOtp({ identifier: row.email, kind: 'cancel_deletion' });

      // Enqueue OTP email via a new kind or reuse 'login-otp' — propose 'cancel-deletion-otp' kind.
      // For minimum surface change, reuse 'login-otp' kind with a payload variant flag.
      // PLANNER NOTE: Plan 06-11 task may add 'cancel-deletion-otp' as a new EmailJobKind + template + worker case.
      await addEmailJob({
        to: '',
        kind: 'cancel-deletion-otp',  // <-- requires Plan 06-11 to add this kind to queue.ts + worker.tsx + template
        userId,
        otpCode,
      });

      return { ok: true };
    }

    // ============================================================================
    // GDPR-05 — Verify cancel-deletion OTP
    // ============================================================================
    export async function verifyCancelDeletionOtp(input: { code: string }): Promise<ActionResult> {
      let userId: string;
      try { userId = await requireUserId(); } catch { return { ok: false, code: 'unauthorized' }; }

      const parsed = cancelDeletionOtpSchema.safeParse(input);
      if (!parsed.success) return { ok: false, code: 'invalid_input' };

      const [row] = await db.select({ status: users.status, email: users.email })
        .from(users).where(eq(users.id, userId)).limit(1);
      if (!row || !row.email) return { ok: false, code: 'wrong_status' };
      if (row.status !== 'pending_deletion') return { ok: false, code: 'wrong_status' };

      // Verify OTP via existing helper
      const ok = await verifyHashedOtp({
        identifier: row.email,
        code: parsed.data.code,
        kind: 'cancel_deletion',
      });
      if (!ok) return { ok: false, code: 'invalid_input', message: 'wrong_code' };

      // Flip status + clear deletion_requested_at
      await db
        .update(users)
        .set({
          status: 'active',
          deletion_requested_at: null,
        })
        .where(eq(users.id, userId));

      // Remove the delayed cascade job (best-effort; log but don't fail action on Redis hiccup)
      try {
        const q = getQueue();
        await q.remove(`account-deletion-cascade:${userId}`);
      } catch {
        // best-effort; Pitfall A safety-net cron would catch a stale job, but in practice
        // a user who cancelled deletion + re-deleted would just hit the rate limit.
      }

      // Send confirmation email
      await addEmailJob({ to: '', kind: 'account-deletion-cancelled', userId });

      revalidatePath('/member');
      revalidatePath('/member/preferences');
      return { ok: true };
    }
    ```
