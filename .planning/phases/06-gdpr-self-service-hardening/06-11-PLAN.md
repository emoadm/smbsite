    ```tsx
    import { redirect } from 'next/navigation';
    import { getTranslations } from 'next-intl/server';
    import { auth } from '@/lib/auth';
    import { db } from '@/db';
    import { users } from '@/db/schema';
    import { eq } from 'drizzle-orm';
    import { Card, CardHeader, CardContent } from '@/components/ui/card';
    import { CancelDeletionVerifyForm } from '@/components/forms/CancelDeletionVerifyForm';

    export const metadata = { title: 'Потвърдете отказването от изтриване' };

    function maskEmail(email: string): string {
      const [local, domain] = email.split('@');
      if (!domain) return email;
      const visible = local.slice(0, 1);
      return `${visible}${'*'.repeat(Math.max(local.length - 1, 1))}@${domain}`;
    }

    export default async function CancelDeletionVerifyPage() {
      const t = await getTranslations('cancelDeletion');
      const session = await auth();
      if (!session?.user?.id) redirect('/auth/login?returnTo=/cancel-deletion/verify');

      const [row] = await db
        .select({ status: users.status, email: users.email })
        .from(users)
        .where(eq(users.id, session.user.id))
        .limit(1);

      if (!row || row.status !== 'pending_deletion' || !row.email) {
        redirect('/account-pending-deletion');
      }

      const maskedEmail = maskEmail(row.email);

      return (
        <div className="container max-w-[480px] mx-auto py-16 px-4">
          <Card>
            <CardHeader>
              <h1 className="font-display text-3xl font-extrabold text-foreground text-balance">
                {t('heading')}
              </h1>
            </CardHeader>
            <CardContent className="space-y-4">
              <p className="text-base leading-normal text-muted-foreground">
                {t('lead', { maskedEmail })}
              </p>
              <CancelDeletionVerifyForm />
            </CardContent>
          </Card>
    ```

    **File 3 — `src/components/forms/CancelDeletionVerifyForm.tsx`** (client component):

    ```tsx
    'use client';

    import { useState, useTransition } from 'react';
    import { useRouter } from 'next/navigation';
    import { useTranslations } from 'next-intl';
    import { Loader2 } from 'lucide-react';
    import { Button } from '@/components/ui/button';
    import { Input } from '@/components/ui/input';
    import { Label } from '@/components/ui/label';
    import { Alert, AlertDescription } from '@/components/ui/alert';
    import { verifyCancelDeletionOtp } from '@/lib/data-rights/actions';

    export function CancelDeletionVerifyForm() {
      const t = useTranslations('cancelDeletion');
      const router = useRouter();
      const [code, setCode] = useState('');
      const [error, setError] = useState<string | null>(null);
      const [isPending, startTransition] = useTransition();

      function handleSubmit(e: React.FormEvent) {
        e.preventDefault();
        setError(null);
        startTransition(async () => {
          const result = await verifyCancelDeletionOtp({ code });
          if (result.ok) {
            router.push('/member?cancel-deletion=success');
          } else if (result.code === 'invalid_input' && result.message === 'wrong_code') {
            setError(t('wrongCode'));
          } else {
            setError(t('wrongCode')); // generic fallback
          }
        });
      }

      return (
        <form onSubmit={handleSubmit} className="space-y-4">
          <div className="space-y-2">
            <Label htmlFor="otp-code">{t('codeLabel')}</Label>
            <Input
              id="otp-code"
              value={code}
              onChange={(e) => setCode(e.target.value)}
              autoComplete="one-time-code"
              inputMode="numeric"
              pattern="\d{6}"
              maxLength={6}
              required
            />
          </div>
          {error && (
            <Alert variant="destructive" role="alert">
              <AlertDescription>{error}</AlertDescription>
            </Alert>
          )}
          <Button type="submit" disabled={code.length !== 6 || isPending} className="w-full min-h-[44px]">
            {isPending && <Loader2 className="size-4 animate-spin mr-2" aria-hidden="true" />}
            {t('submitCta')}
          </Button>
        </form>
      );
    }
    ```
