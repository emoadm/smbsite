    **File 2 — `src/lib/storage/bunny.ts`** (~60 lines)
    ```typescript
    import { createHash } from 'node:crypto';
    import type {
      SignBunnyUrlOptions,
      UploadExportOptions,
      DeleteBunnyBlobOptions,
    } from './bunny.types';

    /**
     * Generate a Bunny.net Token Authentication V2 signed URL.
     *
     * Phase 6 D-01: 7-day TTL on data-export bundles.
     *
     * Algorithm per Bunny docs:
     *   hash_input = security_key + path + expires
     *   token = Base64URL( SHA256_RAW(hash_input) )
     *     Base64URL = standard Base64 with '+' → '-', '/' → '_', '=' stripped (RFC 4648).
     *
     * @returns Full URL with `?token=<b64url>&expires=<unix>` query.
     * @throws Error if neither opts.securityKey nor BUNNY_TOKEN_AUTH_KEY env is set.
     */
    export function signBunnyUrl(opts: SignBunnyUrlOptions): string {
      const securityKey = opts.securityKey ?? process.env.BUNNY_TOKEN_AUTH_KEY;
      if (!securityKey) {
        throw new Error('BUNNY_TOKEN_AUTH_KEY not configured');
      }

      const expires = Math.floor(Date.now() / 1000) + opts.ttlSeconds;
      const hashInput = securityKey + opts.path + String(expires);

      const tokenB64 = createHash('sha256').update(hashInput).digest('base64');
      const tokenUrl = tokenB64
        .replace(/\+/g, '-')
        .replace(/\//g, '_')
        .replace(/=+$/, '');

      return `https://${opts.pullZoneHostname}${opts.path}?token=${tokenUrl}&expires=${expires}`;
    }

    /**
     * PUT a JSON bundle to a Bunny Storage Zone.
     *
     * Phase 6 D-01: worker calls this from the data-export job handler.
     *
     * @throws Error with response status + body on non-2xx.
     * @throws Error if BUNNY_STORAGE_ACCESS_KEY env is not set.
     */
    export async function uploadExportBundle(opts: UploadExportOptions): Promise<void> {
      const accessKey = process.env.BUNNY_STORAGE_ACCESS_KEY;
      if (!accessKey) {
        throw new Error('BUNNY_STORAGE_ACCESS_KEY not configured');
      }

      const url = `https://storage.bunnycdn.com/${opts.storageZone}${opts.path}`;
      const res = await fetch(url, {
        method: 'PUT',
        headers: {
          AccessKey: accessKey,
          'content-type': 'application/json',
        },
        body: opts.body,
      });

      if (!res.ok) {
        const detail = await res.text().catch(() => '<unreadable response>');
        throw new Error(`Bunny storage upload failed: ${res.status} ${detail}`);
      }
    }

    /**
     * DELETE a blob from a Bunny Storage Zone. Idempotent: 404 returns success.
     *
     * Phase 6 D-07 step 6.5: cascade revokes outstanding export bundles.
     *
     * @throws Error on 4xx (other than 404) / 5xx.
     * @throws Error if BUNNY_STORAGE_ACCESS_KEY env is not set.
     */
    export async function deleteBunnyBlob(opts: DeleteBunnyBlobOptions): Promise<void> {
      const accessKey = process.env.BUNNY_STORAGE_ACCESS_KEY;
      if (!accessKey) {
        throw new Error('BUNNY_STORAGE_ACCESS_KEY not configured');
      }

      const url = `https://storage.bunnycdn.com/${opts.storageZone}${opts.path}`;
      const res = await fetch(url, {
        method: 'DELETE',
        headers: { AccessKey: accessKey },
      });

      // 404 = already gone = success (idempotency for cascade re-runs)
      if (res.status === 404) return;

      if (!res.ok) {
        const detail = await res.text().catch(() => '<unreadable response>');
        throw new Error(`Bunny storage delete failed: ${res.status} ${detail}`);
      }
    }
    ```
