    ```typescript
    import { createHash } from 'node:crypto';
    import { eq } from 'drizzle-orm';
    import { db } from '@/db';  // confirm exact path; existing imports use this style

      users,
      consents,
      attribution_events,
      submissions,

    } from '@/db/schema';
    // newsletter_preferences import path — verify against actual Phase 5 schema file location

    export const EMPTY_BUNDLE_TABLE_KEYS = [
      'users',
      'votes',
      'submissions',
      'consents',
      'attribution_events',
      'newsletter_preferences',
      'moderation_log',
    ] as const;

    export interface ExportBundleV1 {
      metadata: {
        exported_at: string;       // ISO 8601 UTC
        format_version: 1;
        user_id_hash: string;      // sha256 hex of the original user.id
      };
      users: unknown[];
      votes: unknown[];
      submissions: unknown[];
      consents: unknown[];
      attribution_events: unknown[];
      newsletter_preferences: unknown[];
      moderation_log: unknown[];
    }

    /**
     * Build the GDPR-04 data-export bundle for a single user.
     *
     * D-02 scope: 7 tables. Stable schema: empty arrays render as [] not null.
     * D-03 format: caller serializes with JSON.stringify(_, null, 2).
     *
     * Performance: one SELECT per table; no N+1. Total bundle < 100KB for typical users.
     */
    export async function buildExportBundle(userId: string): Promise<ExportBundleV1> {
      const userIdHash = createHash('sha256').update(userId).digest('hex');

      const [userRow] = await db.select().from(users).where(eq(users.id, userId)).limit(1);

      // votes: empty until Phase 3 ships voting schema. Return [] (stable shape).
      const votesRows: unknown[] = [];

      const submissionsRows = await db
        .select()
        .from(submissions)
        .where(eq(submissions.submitter_id, userId));  // verify column name

      const consentsRows = await db
        .select()
        .from(consents)
        .where(eq(consents.user_id, userId));  // verify column name

      const attributionEventsRows = await db
        .select()
        .from(attribution_events)
        .where(eq(attribution_events.user_id, userId));

      // newsletter_preferences read — verify table import + column name.
      // If table not yet imported here, leave as [] with a TODO comment for Plan 06-06 to wire.
      const newsletterPreferencesRows: unknown[] = []; // TODO: wire to newsletter_preferences table from Phase 5

      const moderationLogRows = await db
        .select()
        .from(moderation_log)
        .where(eq(moderation_log.actor_user_id, userId));  // verify column name

      return {
        metadata: {
          exported_at: new Date().toISOString(),
          format_version: 1,
          user_id_hash: userIdHash,
        },
        users: userRow ? [userRow] : [],
        votes: votesRows,
        submissions: submissionsRows,
        consents: consentsRows,
        attribution_events: attributionEventsRows,
        newsletter_preferences: newsletterPreferencesRows,
        moderation_log: moderationLogRows,
      };
    }
    ```

    ```tsx
    case 'data-export': {
      const userId = job.data.userId;
      if (!userId) throw new Error('data-export job missing userId');

      const storageZone = process.env.BUNNY_STORAGE_ZONE;
      const pullZoneHostname = process.env.BUNNY_EXPORT_HOSTNAME;
      if (!storageZone || !pullZoneHostname) {
        throw new Error('Bunny export env not configured (BUNNY_STORAGE_ZONE / BUNNY_EXPORT_HOSTNAME)');
      }

      const bundle = await buildExportBundle(userId);
      const body = Buffer.from(JSON.stringify(bundle, null, 2), 'utf8');

      const userIdHash16 = createHash('sha256').update(userId).digest('hex').slice(0, 16);
      const filename = `${Date.now()}-${randomUUID()}.json`;
      const bunnyPath = `/exports/${userIdHash16}/${filename}`;

      await uploadExportBundle({ storageZone, path: bunnyPath, body });

      const ttlSeconds = 7 * 86_400;
      const expiresAtSeconds = Math.floor(Date.now() / 1000) + ttlSeconds;
      const exportUrl = signBunnyUrl({ pullZoneHostname, path: bunnyPath, ttlSeconds });
      const expiresAt = new Date(expiresAtSeconds * 1000);

      // Track for D-07 step 6.5 revocation enumeration.
      await db.insert(data_export_jobs).values({
        user_id: userId,
        bunny_path: bunnyPath,
        expires_at: expiresAt,
      });

      // Enqueue the notification email
      await addEmailJob({
        to: '',
        kind: 'data-export-ready',
        userId,
        exportUrl,
        exportPath: bunnyPath,
        expiresAt,
      });

      return { messageId: `data-export-prepared-${bunnyPath}` };
    }

    // ============================================================================
    // Phase 6 D-01 — data-export-ready notification email.
    // Pattern P6: render(_, { plainText: true }) for textContent (Quick 260514-k4x lock).
    // ============================================================================
    case 'data-export-ready': {
      const userId = job.data.userId;
      const exportUrl = job.data.exportUrl;
      const expiresAtIso = job.data.expiresAt;
      if (!userId || !exportUrl || !expiresAtIso) {
        throw new Error('data-export-ready payload missing userId/exportUrl/expiresAt');
      }

      const [row] = await db
        .select({ email: users.email, full_name: users.full_name })
        .from(users)
        .where(eq(users.id, userId))
        .limit(1);

      if (!row || !row.email) {
        return { messageId: 'skipped-missing-row' };
      }

      const tEmail = loadT('email.dataExportReady');
      const expiresAt = new Date(expiresAtIso);
      const emailElement = (
        <DataExportReadyEmail
          t={tEmail}
          fullName={row.full_name}
          exportUrl={exportUrl}
          expiresAt={expiresAt}
        />
      );
      const emailHtml = await render(emailElement);
      const emailText = await render(emailElement, { plainText: true });

      return sendBrevoEmail({
        to: { email: row.email, name: row.full_name },
        subject: tEmail('subject'),
        htmlContent: emailHtml,
        textContent: emailText,
        from: {
          email: process.env.EMAIL_FROM_TRANSACTIONAL ?? 'no-reply@auth.chastnik.eu',
          name: loadT('email.from')('name'),
        },
      });
    }
    ```
