    ```typescript
     * Idempotent: 404 (already gone) returns success.

    export async function deleteBrevoContact(email: string): Promise<void> {
      const apiKey = process.env.BREVO_API_KEY;
      if (!apiKey) throw new Error('BREVO_API_KEY not configured');

      const res = await fetch(`https://api.brevo.com/v3/contacts/${encodeURIComponent(email)}`, {
        method: 'DELETE',
        headers: { 'api-key': apiKey, accept: 'application/json' },
      });
      if (res.status === 404) return;  // already gone
      if (!res.ok) {
        const body = await res.text().catch(() => '<unreadable>');
        throw new Error(`Brevo contact delete failed: ${res.status} ${body}`);
      }
    }
    ```

    **Step 3 — Author `src/lib/data-rights/cascade.ts`**:

    ```typescript
    import { createHash } from 'node:crypto';
    import { and, eq } from 'drizzle-orm';
    import { db } from '@/db';
    import {
      users,
      attribution_events,
      consents,
      sessions,
      deletion_log,
      data_export_jobs,
    } from '@/db/schema';
    // newsletter_preferences import — verify path; see Plan 06-05 note.
    import { deleteBunnyBlob } from '@/lib/storage/bunny';
    import { deleteBrevoContact } from '@/lib/email/brevo';
    import { addEmailJob } from '@/lib/email/queue';
    import { getQueue } from '@/lib/email/queue'; // for removeJobs in step 6
    import pino from 'pino';

    const log = pino({ name: 'deletion-cascade' });

    /**
     * Phase 6 D-07 — 9-step account deletion cascade.
     *
     * Wraps steps 1-5 + 8 in a single Drizzle transaction.
     * Steps 6, 6.5, and 7 run OUTSIDE the tx (Pitfall B — external API in tx = bad).
     *
     * Idempotent: re-running on an already-deleted user is a no-op
     * (step 1 UPDATE WHERE status='pending_deletion' affects 0 rows → throws 'alreadyHandled').
     *
     * Caller (worker handler): catch 'alreadyHandled' → log + return without retry.
     * All other errors → bubble up → BullMQ retries per its retry policy.
     */
    export async function runDeletionCascade(userId: string): Promise<void> {
      // Pre-cascade check: read the user; refuse if not in pending_deletion or grace not elapsed.
      const [row] = await db
        .select({
          id: users.id,
          email: users.email,
          status: users.status,
          deletion_requested_at: users.deletion_requested_at,
        })
        .from(users)
        .where(eq(users.id, userId))
        .limit(1);

      if (!row) {
        log.warn({ userId }, 'cascade-skip: user not found');
        return;
      }
      if (row.status !== 'pending_deletion') {
        log.warn({ userId, status: row.status }, 'cascade-skip: user not in pending_deletion');
        return;
      }
      if (!row.deletion_requested_at) {
        log.warn({ userId }, 'cascade-skip: deletion_requested_at missing');
        return;
      }
      const graceEndsAt = new Date(row.deletion_requested_at.getTime() + 30 * 86_400_000);
      if (Date.now() < graceEndsAt.getTime()) {
        log.warn({ userId, graceEndsAt }, 'cascade-skip: grace period not yet expired');
        return;
      }

      const originalEmail = row.email;

      // Pre-fetch data_export_jobs paths so we can DELETE the Bunny blobs after tx commits.
      const exportJobRows = await db
        .select({ id: data_export_jobs.id, bunny_path: data_export_jobs.bunny_path })
        .from(data_export_jobs)
        .where(eq(data_export_jobs.user_id, userId));

      // === TRANSACTION: steps 1-5 + 8 ===
      await db.transaction(async (tx) => {
        // Step 1: tombstone users row (UPDATE, never DELETE — see Pitfall C)
        const updated = await tx
          .update(users)
          .set({
            email: null,
            full_name: null,
            // phone: null,  // verify column exists in users schema; if not, omit
            // oblast: null, // verify column exists; if not, omit
            self_reported_source: null,
            self_reported_other: null,
            status: 'deleted',
            // email_hash retained per D-06 — do NOT null this out
          })
          .where(and(eq(users.id, userId), eq(users.status, 'pending_deletion')))
          .returning({ id: users.id });

        if (updated.length === 0) throw new Error('alreadyHandled');

        // Step 2: attribution_events — unlink user_id, preserve aggregate counts
        await tx
          .update(attribution_events)
          .set({ user_id: null })
          .where(eq(attribution_events.user_id, userId));

        // Step 3: consents — strip region + email_at_consent (retain user_id_hash if present)
        await tx
          .update(consents)
          .set({ region: null, email_at_consent: null })
          .where(eq(consents.user_id, userId));

        // Step 4: sessions — forced logout on all devices
        await tx.delete(sessions).where(eq(sessions.userId, userId));

        // Step 5: newsletter_preferences — hard delete (no retention purpose)
        // await tx.delete(newsletter_preferences).where(eq(newsletter_preferences.user_id, userId));
        // (Wire when newsletter_preferences import is confirmed — see Plan 06-05 NOTE.)

        // Step 6.5 — DELETE the data_export_jobs rows inside the tx so the cascade
        // is durable; the Bunny blob DELETE happens OUTSIDE the tx (network call).
        // (Section 7 REVOKE allows DELETE on data_export_jobs but blocks UPDATE.)
        if (exportJobRows.length > 0) {
          await tx
            .delete(data_export_jobs)
            .where(eq(data_export_jobs.user_id, userId));
        }

        // Step 8: append-only deletion_log INSERT — only the hash + reason
        await tx.insert(deletion_log).values({
          target_user_id_hash: createHash('sha256').update(userId).digest('hex'),
          deleted_at: new Date(),
          deletion_reason: 'user_request',
        });
      });

      log.info({ userId, tombstone: true, exportBlobs: exportJobRows.length }, 'cascade-tx-committed');

      // === POST-TX (external + best-effort, retried if they throw) ===

      // Step 6 — Cancel any in-flight BullMQ jobs for this userId.
      // BullMQ removeJobs is best-effort; if Redis is unavailable we log + continue
      // (the next phase reconciles via the typical job retry path).
      try {
        const q = getQueue();
        // Remove all jobs whose data.userId matches. Pattern adapted from Phase 5
        // unsubscribe pattern; BullMQ doesn't expose direct querying by data, so we
        // use job IDs convention if we stamp them: `account-deletion-cascade:${userId}`.
        // (This requires Plan 06-10 requestDeletion Server Action to set jobId.)
        await q.remove(`account-deletion-cascade:${userId}`);
      } catch (e) {
        log.warn({ userId, err: String(e) }, 'cascade-step6-bullmq-remove-failed-noncritical');
      }

      // Step 6.5 — DELETE Bunny blobs for outstanding exports (idempotent 404-tolerant).
      const storageZone = process.env.BUNNY_STORAGE_ZONE;
      if (storageZone) {
        for (const exp of exportJobRows) {
          try {
            await deleteBunnyBlob({ storageZone, path: exp.bunny_path });
          } catch (e) {
            log.warn({ userId, path: exp.bunny_path, err: String(e) }, 'cascade-step6.5-bunny-delete-failed');
            // Don't throw — the blob remains in Bunny but the signed URL is now stale
            // (the signing key + expiry are still valid; revocation is incomplete).
            // Operator follow-up via Bunny dashboard if persistent.
          }
        }
      }

      // Step 7 — Brevo DELETE /contacts/{email} — idempotent, retry-safe.
      if (originalEmail) {
        try {
          await deleteBrevoContact(originalEmail);
          log.info({ userId }, 'cascade-step7-brevo-deleted');
        } catch (e) {
          log.warn({ userId, err: String(e) }, 'cascade-step7-brevo-failed-enqueueing-retry');
          // Enqueue a retry job (idempotent)
          await addEmailJob({
            to: '',
            kind: 'brevo-contact-delete-retry',
            unsubEmail: originalEmail,
          });
        }
      }
    }
    ```
