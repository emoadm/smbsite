    ```tsx
    'use client';

    import { useState, useTransition } from 'react';
    import { useRouter } from 'next/navigation';
    import { useTranslations } from 'next-intl';
    import { Loader2 } from 'lucide-react';
    import {
      Dialog, DialogTrigger, DialogContent, DialogHeader, DialogTitle,
      DialogDescription, DialogFooter,
    } from '@/components/ui/dialog';
    import { Button } from '@/components/ui/button';
    import { Input } from '@/components/ui/input';
    import { Label } from '@/components/ui/label';
    import { Separator } from '@/components/ui/separator';
    import { Alert, AlertDescription } from '@/components/ui/alert';
    import { requestDeletion } from '@/lib/data-rights/actions';
    import { TYPED_DELETE_CONFIRM } from '@/lib/data-rights/zod';

    export function DeleteAccountDialog() {
      const t = useTranslations('member.deleteAccount');
      const tCard = useTranslations('member.preferences.dataRights');
      const router = useRouter();
      const [open, setOpen] = useState(false);
      const [typedValue, setTypedValue] = useState('');
      const [isPending, startTransition] = useTransition();
      const [submitError, setSubmitError] = useState(false);

      const matches = typedValue.trim() === TYPED_DELETE_CONFIRM;
      const canConfirm = matches && !isPending;

      function handleSubmit() {
        setSubmitError(false);
        startTransition(async () => {
          const result = await requestDeletion({ typedConfirm: typedValue });
          if (result.ok) {
            setOpen(false);
            router.push('/account-pending-deletion');
          } else {
            setSubmitError(true);
          }
        });
      }

      // Reset state on close
      function handleOpenChange(next: boolean) {
        if (isPending) return; // non-dismissible during submit
        setOpen(next);
        if (!next) {
          setTypedValue('');
          setSubmitError(false);
        }
      }

      return (
        <Dialog open={open} onOpenChange={handleOpenChange}>
          <DialogTrigger asChild>
            <Button variant="destructive" size="lg" className="min-h-[44px]">
              {tCard('delete.cta')}
            </Button>
          </DialogTrigger>
          <DialogContent
            className="sm:max-w-lg"
            showCloseButton={!isPending}
            onEscapeKeyDown={(e) => { if (isPending) e.preventDefault(); }}
            onPointerDownOutside={(e) => { if (isPending) e.preventDefault(); }}
            onOpenAutoFocus={(e) => {
              e.preventDefault();
              setTimeout(() => document.getElementById('delete-confirm-input')?.focus(), 0);
            }}
            aria-busy={isPending}
          >
            <DialogHeader>
              <DialogTitle className="text-destructive">{t('title')}</DialogTitle>
              <DialogDescription>{t('description')}</DialogDescription>
            </DialogHeader>

            <ul role="list" className="space-y-2 mt-2 text-base list-disc pl-5">
              {[0, 1, 2, 3, 4].map((i) => (
                <li key={i}>{t(`bullets.${i}`)}</li>
              ))}
            </ul>

            <Separator className="my-4" />

            <div className="space-y-2">
              <Label htmlFor="delete-confirm-input">
                {t.rich('typedConfirmLabel', {
                  code: (chunks) => <code className="font-mono font-semibold">{chunks}</code>,
                })}
              </Label>
              <Input
                id="delete-confirm-input"
                value={typedValue}
                onChange={(e) => setTypedValue(e.target.value)}
                autoComplete="off"
                autoCorrect="off"
                autoCapitalize="off"
                spellCheck={false}
                inputMode="text"
                aria-describedby="delete-confirm-help"
              />
              <p id="delete-confirm-help" className="text-xs leading-normal text-muted-foreground">
                {t('typedConfirmHelp')}
              </p>
            </div>

            {submitError && (
              <Alert variant="destructive" role="alert" className="mt-4">
                <AlertDescription>{t('submitError')}</AlertDescription>
              </Alert>
            )}

            <DialogFooter className="mt-4 flex gap-3 justify-end">
              <Button variant="outline" onClick={() => handleOpenChange(false)} disabled={isPending}>
                {t('dismissCta')}
              </Button>
              <Button
                variant="destructive"
                className="min-h-[44px]"
                disabled={!canConfirm}
                aria-disabled={!canConfirm}
                onClick={handleSubmit}
              >
                {isPending && <Loader2 className="size-4 animate-spin mr-2" aria-hidden="true" />}
                {t('confirmCta')}
              </Button>
            </DialogFooter>
          </DialogContent>
        </Dialog>
      );
    }
    ```

    **File 2 — `src/app/(frontend)/account-pending-deletion/page.tsx`**:

    ```tsx
    import { redirect } from 'next/navigation';
    import { getTranslations } from 'next-intl/server';
    import { signOut } from '@/lib/auth'; // verify export path
    import { auth } from '@/lib/auth';
    import { db } from '@/db';
    import { users } from '@/db/schema';
    import { eq } from 'drizzle-orm';
    import { Card, CardHeader, CardContent } from '@/components/ui/card';
    import { Alert, AlertDescription } from '@/components/ui/alert';
    import { Button } from '@/components/ui/button';
    import { requestCancelDeletionOtp } from '@/lib/data-rights/actions';

    export const metadata = { title: 'Изтриване насрочено' };

    async function cancelDeletionAction() {
      'use server';
      const result = await requestCancelDeletionOtp({});
      if (!result.ok) {
        redirect('/account-pending-deletion?error=cancel-failed');
      }
      redirect('/cancel-deletion/verify');
    }

    async function logoutAction() {
      'use server';
      await signOut({ redirectTo: '/' });
    }

    interface PageProps {
      searchParams: Promise<{ error?: string }>;
    }

    export default async function AccountPendingDeletionPage({ searchParams }: PageProps) {
      const t = await getTranslations('accountPendingDeletion');
      const session = await auth();
      if (!session?.user?.id) redirect('/auth/login?returnTo=/account-pending-deletion');

      const [row] = await db
        .select({ status: users.status, deletion_requested_at: users.deletion_requested_at })
        .from(users)
        .where(eq(users.id, session.user.id))
        .limit(1);

      // User is not pending → bounce to /member
      if (!row || row.status !== 'pending_deletion' || !row.deletion_requested_at) {
        redirect('/member');
      }

      const graceEndsAt = new Date(row.deletion_requested_at.getTime() + 30 * 86_400_000);
      const dateFormatted = graceEndsAt.toLocaleDateString('bg-BG', {
        day: '2-digit', month: '2-digit', year: 'numeric',
      });

      const params = await searchParams;
      const showError = params.error === 'cancel-failed';

      return (
        <div className="container max-w-[480px] mx-auto py-16 px-4">
          <Card>
            <CardHeader>
              <h1 className="font-display text-3xl font-extrabold text-destructive text-balance">
                {t('heading')}
              </h1>
            </CardHeader>
            <CardContent className="space-y-4">
              <p
                className="text-base leading-normal"
                dangerouslySetInnerHTML={{
                  __html: t.markup('body', {
                    date: dateFormatted,
                    strong: (chunks) => `<strong class="font-feature-settings-tnum">${chunks}</strong>`,
                  }),
                }}
              />
              <p className="text-base leading-normal text-muted-foreground">{t('consequences')}</p>
    ```
