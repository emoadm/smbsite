    import { pgTable, uuid, text, timestamp, index } from 'drizzle-orm/pg-core';

    // Phase 6 D-07 / D-08 — INSERT-only audit log of account deletions.
    // GDPR-08 contract: no PII columns. target_user_id_hash is SHA-256 hex
    // of the original user.id (irreversible to PII because user.id is itself
    // a non-PII random UUID; the hash exists to prevent accidental linkage
    // through the original UUID surfacing elsewhere).
    // REVOKE UPDATE, DELETE enforced at DB level (see 0004 migration).
    export const deletion_log = pgTable(
      'deletion_log',
      {
        id: uuid('id').defaultRandom().primaryKey(),
        target_user_id_hash: text('target_user_id_hash').notNull(),
        deleted_at: timestamp('deleted_at', { withTimezone: true }).defaultNow().notNull(),
        deletion_reason: text('deletion_reason').notNull(), // 'user_request' (extend later if needed)
      },
      (t) => ({

    // Phase 6 D-09 — Anonymous cookie-consent persistence.
    // Carry-forward from Phase 1 plan-check N-1. Linked to users.id post-OTP-linkage
    // (Plan 06-09 Server Action triggers the UPDATE when a user with an attr_sid registers).
    // REVOKE DELETE at DB level — consent history is auditable. UPDATE remains so user_id

      'cookie_consents',
      {
        id: uuid('id').defaultRandom().primaryKey(),
        anon_id: uuid('anon_id').notNull(),
        user_id: uuid('user_id').references(() => users.id, { onDelete: 'set null' }),
        choices: jsonb('choices').notNull(),
        consent_at: timestamp('consent_at', { withTimezone: true }).defaultNow().notNull(),
        withdrawn_at: timestamp('withdrawn_at', { withTimezone: true }),
        ip_hash: text('ip_hash'),
        region: text('region'),
