GitHub Actions step pattern (existing pattern in deploy.yml — read for shape):
```yaml
- name: Cloudflare cache purge
  if: success()
  continue-on-error: true
  run: |
    response=$(curl -sS -w '\n%{http_code}' -X POST \
      "https://api.cloudflare.com/client/v4/zones/${{ secrets.CF_ZONE_ID }}/purge_cache" \
      -H "Authorization: Bearer ${{ secrets.CF_API_TOKEN }}" \
      -H "Content-Type: application/json" \
      -d '{"purge_everything": true}')
    body=$(echo "$response" | head -n -1)
    code=$(echo "$response" | tail -n 1)
    if [ "$code" != "200" ]; then
      echo "::warning::Cloudflare cache purge failed ($code): $body"
      echo "## ⚠ Cloudflare cache purge skipped — manual purge may be required" >> $GITHUB_STEP_SUMMARY
      exit 0  # don't fail the deploy
    fi
    echo "Cloudflare purge succeeded"
```
