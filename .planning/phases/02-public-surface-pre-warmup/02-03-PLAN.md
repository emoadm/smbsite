    ```tsx
    'use client';

    import { useEffect, useState } from 'react';

    /**
     * Sticky-on-md+ list of section h2 anchors. On mobile collapses to <details>.
     * UI-SPEC §5.3 — agenda page only. Items are passed in as props
     * (auto-generation from MDX headings is deferred).
     */
    export function TableOfContents({
      items,
      label,
    }: {
      items: { id: string; label: string }[];
      label: string;
    }) {
      // Track active section for active-state styling (optional polish; non-blocking)
      const [activeId, setActiveId] = useState<string>(items[0]?.id ?? '');

      useEffect(() => {
        if (typeof window === 'undefined') return;
        const observer = new IntersectionObserver(
          (entries) => {
            const visible = entries
              .filter((e) => e.isIntersecting)
              .sort((a, b) => a.boundingClientRect.top - b.boundingClientRect.top)[0];
            if (visible) setActiveId(visible.target.id);
          },
          { rootMargin: '-30% 0px -60% 0px' },
        );
        items.forEach((item) => {
          const el = document.getElementById(item.id);
          if (el) observer.observe(el);
        });
        return () => observer.disconnect();
      }, [items]);

      return (
        <>
          {/* Mobile: collapsible <details> */}
          <details className="md:hidden mb-6 rounded-md border border-border bg-surface p-4">
            <summary className="font-display text-base">{label}</summary>
            <ul className="mt-3 space-y-2 text-sm">
              {items.map((item) => (
                <li key={item.id}>
                  <a href={`#${item.id}`} className="text-primary hover:underline">
                    {item.label}
                  </a>
                </li>
              ))}
            </ul>
          </details>
          {/* Desktop: sticky sidebar */}
          <nav
            aria-label={label}
            className="hidden md:block sticky top-20 self-start"
          >
            <ul className="space-y-2 text-sm">
              {items.map((item) => (
                <li key={item.id}>
                  <a
                    href={`#${item.id}`}
                    className={
                      item.id === activeId
                        ? 'font-display text-primary'
                        : 'text-muted-foreground hover:text-primary'
                    }
                  >
                    {item.label}
                  </a>
                </li>
              ))}
            </ul>
          </nav>
        </>
      );
    }
    ```

    - IntersectionObserver is wrapped in `if (typeof window === 'undefined')` for SSR safety.
