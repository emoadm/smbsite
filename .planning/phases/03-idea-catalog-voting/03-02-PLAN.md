```typescript
// Lazy-init: build must not fail when secret unset. First sign call throws.
// Mirrors src/lib/unsubscribe/hmac.ts:24-28.
function SECRET(): string {
  const s = process.env.VOTE_AUDIT_HMAC_SECRET;
  if (!s) throw new Error('VOTE_AUDIT_HMAC_SECRET not set');
  return s;
}

export function hashIp(ip: string): string {
  return createHmac('sha256', SECRET()).update(ip).digest('hex');
}

function subnetPrefix(ip: string): string {
  // IPv4: /24 via existing getSubnet. IPv6: /64.
  if (ip.includes(':')) return getSubnetV6(ip);
  return getSubnet(ip);
}

export function hashSubnet(ip: string): string {
  return createHmac('sha256', SECRET()).update(subnetPrefix(ip)).digest('hex');
}

export function hashUa(ua: string): string {
  return createHmac('sha256', SECRET()).update(ua).digest('hex');
}
```

**File 2: extend `src/lib/ip.ts`** — add `getSubnetV6` for IPv6 /64 prefix. READ existing file first; APPEND new export. Keep existing `getSubnet` and `getClientIp` untouched.

```typescript
// IPv6 /64 — first 4 hextets. Used by hashSubnet for D-15 cluster detection.
// Examples:
//   2001:db8:abcd:0012::1 → 2001:db8:abcd:0012::/64
//   ::1 → ::/64 (loopback)
export function getSubnetV6(ip: string): string {
  if (!ip.includes(':')) return ip;
  // Normalize: expand :: shorthand to a flat 8-hextet representation, then keep first 4.
  const parts = ip.split('::');
  let head: string[] = parts[0] ? parts[0].split(':') : [];
  const tail: string[] = parts[1] ? parts[1].split(':') : [];
  const missing = 8 - head.length - tail.length;
  const expanded = [...head, ...Array(Math.max(0, missing)).fill('0'), ...tail];
  if (expanded.length < 4) return ip; // malformed input — return as-is
  return `${expanded.slice(0, 4).join(':')}::/64`;
}
```

```typescript
// D-10 — VOTE_COUNTABLE_INTERVAL is a Postgres INTERVAL literal validated at boot.
// Allowlisted regex defeats SQL injection (sql.raw() is unavoidable since INTERVAL cannot be parameterized).
export function parseInterval(raw: string): string {
  if (!/^\d+ (hours|days)$/.test(raw)) {
    throw new Error(`Invalid VOTE_COUNTABLE_INTERVAL: ${raw}. Expected /^\\d+ (hours|days)$/`);
  }
  return raw;
}

// Resolved at module load → fails fast on misconfigured deploy.
export const COOLING_INTERVAL = parseInterval(process.env.VOTE_COUNTABLE_INTERVAL ?? '48 hours');

// D-04 — cooled approve count: vote rows JOIN users.email_verified_at + INTERVAL filter.
// Implementation note: sql.raw is safe because COOLING_INTERVAL is regex-whitelisted at boot.
export async function getCooledApproveCount(db: typeof DrizzleDb, ideaId: string): Promise<number> {
  const result = await db.execute(sql`
    SELECT COUNT(*)::int AS approve_count
    FROM votes v
    INNER JOIN users u ON u.id = v.user_id
    WHERE v.idea_id = ${ideaId}
      AND v.choice = 'approve'
      AND u.email_verified_at + INTERVAL '${sql.raw(COOLING_INTERVAL)}' < NOW()
  `);
  return Number((result.rows[0] as { approve_count?: number })?.approve_count ?? 0);
}

export async function getCooledRejectCount(db: typeof DrizzleDb, ideaId: string): Promise<number> {
  const result = await db.execute(sql`
    SELECT COUNT(*)::int AS reject_count
    FROM votes v
    INNER JOIN users u ON u.id = v.user_id
    WHERE v.idea_id = ${ideaId}
      AND v.choice = 'reject'
      AND u.email_verified_at + INTERVAL '${sql.raw(COOLING_INTERVAL)}' < NOW()
  `);
  return Number((result.rows[0] as { reject_count?: number })?.reject_count ?? 0);
}

// Used at vote-INSERT time to compute fresh_account_at_event boolean (D-15).
// Returns true when the voter's email_verified_at is within the cooling window.
export async function isFreshAccount(db: typeof DrizzleDb, userId: string): Promise<boolean> {
  const result = await db.execute(sql`
    SELECT (NOW() - email_verified_at) < INTERVAL '${sql.raw(COOLING_INTERVAL)}' AS is_fresh
    FROM users WHERE id = ${userId}
  `);
  return Boolean((result.rows[0] as { is_fresh?: boolean })?.is_fresh);
}
```

```typescript
const REVEAL_THRESHOLD = Number(process.env.IDEA_REVEAL_THRESHOLD ?? 20);

export type DisplayCounts =
  | { revealed: false; total: number; remaining: number }
  | { revealed: true; approve: number; reject: number; total: number; approvePct: number };

// D-03 — 5-minute server-side cache; vote write does NOT bust the cache (Pitfall 7).
// Per-pod cache; drift between pods bounded by 5-min TTL.
export const getCachedDisplayCounts = unstable_cache(
  async (ideaId: string): Promise<DisplayCounts> => {
    const [approve, reject] = await Promise.all([
      getCooledApproveCount(db, ideaId),
      getCooledRejectCount(db, ideaId),
    ]);
    const total = approve + reject;
    if (total < REVEAL_THRESHOLD) {
      return { revealed: false, total, remaining: Math.max(1, REVEAL_THRESHOLD - total) };
    }
    return {
      revealed: true,
      approve,
      reject,
      total,
      approvePct: Math.round((approve / total) * 100),
    };
  },
  ['idea-display-counts'],
  { revalidate: 300 },
);
```

```typescript
import { transliterate } from 'transliteration';
import { eq } from 'drizzle-orm';
import { db } from '@/db';
import { ideas } from '@/db/schema/voting';

// 2006 Transliteration Act amendment — word-final -ия → -ia (Sofia, Bulgaria).
// Default `transliteration` mapping covers the rest of the Streamlined System.
const BULGARIAN_OVERRIDES: Array<[RegExp, string]> = [
  [/ия\b/g, 'ia'],
];

function applyOverrides(input: string): string {
  let result = input;
  for (const [pattern, replacement] of BULGARIAN_OVERRIDES) {
    result = result.replace(pattern, replacement);
  }
  return result;
}

export function slugifyBg(title: string): string {
  const overridden = applyOverrides(title);
  return transliterate(overridden, { trim: true })
    .toLowerCase()
    .replace(/[^a-z0-9\s-]/g, '')
    .replace(/\s+/g, '-')
    .replace(/-+/g, '-')
    .replace(/^-+|-+$/g, '');
}

// D-18 + RESEARCH §Pitfall 5 — slug uniqueness via numeric-suffix retry.
export async function ensureUniqueSlug(baseSlug: string): Promise<string> {
  let slug = baseSlug;
  let n = 2;
  // Bound iterations to defend against pathological inputs.
  for (let i = 0; i < 1000; i++) {
    const existing = await db.select({ id: ideas.id }).from(ideas).where(eq(ideas.slug, slug)).limit(1);
    if (existing.length === 0) return slug;
    slug = `${baseSlug}-${n++}`;
  }
  throw new Error(`Could not find unique slug for ${baseSlug} after 1000 tries`);
}
```
