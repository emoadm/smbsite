```tsx
import Link from 'next/link';
import { Clock } from 'lucide-react';
import { getTranslations } from 'next-intl/server';
import { Card, CardContent } from '@/components/ui/card';
import { Badge } from '@/components/ui/badge';
import { Button } from '@/components/ui/button';
import { db } from '@/db';
import { votes } from '@/db/schema/voting';
import { users } from '@/db/schema/auth';
import { sql, eq, desc } from 'drizzle-orm';
import { getCachedDisplayCounts } from '@/lib/voting/cache';
import { CoolingIndicator } from './CoolingIndicator';
import { MyActivityVoteActions } from './MyActivityVoteActions';

const COOLING_INTERVAL_MS = (() => {
  const raw = process.env.VOTE_COUNTABLE_INTERVAL ?? '48 hours';
  const m = raw.match(/^(\d+)\s+(hours|days)$/);
  if (!m) return 48 * 3600 * 1000;
  const n = Number(m[1]);
  return m[2] === 'days' ? n * 86400 * 1000 : n * 3600 * 1000;
})();

export async function MyActivityPanel({ userId }: { userId: string }) {
  const t = await getTranslations('member.activity');

  // Fetch user's email_verified_at (cooling baseline)
  const [me] = await db.select({ email_verified_at: users.email_verified_at }).from(users).where(eq(users.id, userId)).limit(1);
  const verifiedAt = me?.email_verified_at?.getTime() ?? Date.now();

  // Join votes ↔ ideas (ideas table is Payload-managed; raw SQL OK because no user input)
  const myVotes = await db.execute(sql`
    SELECT v.id AS vote_id, v.choice, v.created_at, v.idea_id, i.title, i.slug
    FROM votes v
    INNER JOIN ideas i ON i.id = v.idea_id
    WHERE v.user_id = ${userId}
    ORDER BY v.created_at DESC
  `);
  const rows = myVotes.rows as Array<{ vote_id: string; choice: 'approve' | 'reject'; created_at: Date; idea_id: string; title: string; slug: string }>;

  if (rows.length === 0) {
    return (
      <section className="mt-12">
        <h2 className="font-display text-2xl">{t('title')}</h2>
        <p className="text-base text-muted-foreground">{t('subtitle')}</p>
        <div className="mt-6 text-center py-12">
          <p className="text-base text-muted-foreground">{t('empty')}</p>
          <Button asChild className="mt-4"><Link href="/idei">{t('emptyCta')}</Link></Button>
        </div>
      </section>
    );
  }

  // Resolve cached counts in parallel
  const counts = await Promise.all(rows.map((r) => getCachedDisplayCounts(r.idea_id)));

  return (
    <section className="mt-12">
      <h2 className="font-display text-2xl">{t('title')}</h2>
      <p className="text-base text-muted-foreground">{t('subtitle')}</p>
      <ol className="mt-6 space-y-4">
        {rows.map((row, idx) => {
          const c = counts[idx];
          const coolingExpiresAt = verifiedAt + COOLING_INTERVAL_MS;
          const isCooling = Date.now() < coolingExpiresAt;
          return (
            <li key={row.vote_id}>
              <Card>
                <CardContent className="flex flex-col gap-3 py-6">
                  <Link href={`/idei/${row.slug}`} className="font-display text-xl text-foreground hover:text-primary">
                    {row.title}
                  </Link>
                  {isCooling && (
                    <CoolingIndicator coolingExpiresAt={coolingExpiresAt} />
                  )}
                  <Badge className="self-start rounded-full bg-primary text-primary-foreground">
                    {row.choice === 'approve' ? 'Одобрено от теб' : 'Не одобрено от теб'}
                  </Badge>
                  {c.revealed && (
                    <p className="text-sm text-muted-foreground [font-feature-settings:'tnum']">
                      {c.approve} одобряват · {c.approvePct}% одобрение
                    </p>
                  )}
                  {!c.revealed && (
                    <p className="text-sm text-muted-foreground italic">Резултатът все още не се показва.</p>
                  )}
                  <MyActivityVoteActions ideaId={row.idea_id} currentChoice={row.choice} />
                </CardContent>
              </Card>
            </li>
          );
        })}
      </ol>
    </section>
  );
}
```

**File 2: src/components/member/CoolingIndicator.tsx** (client island)

```tsx
'use client';
import { useEffect, useState } from 'react';
import { Clock } from 'lucide-react';
import { useTranslations } from 'next-intl';

export function CoolingIndicator({ coolingExpiresAt }: { coolingExpiresAt: number }) {
  const t = useTranslations('idea');
  const [now, setNow] = useState(Date.now());
  useEffect(() => {
    const id = setInterval(() => setNow(Date.now()), 60_000);
    return () => clearInterval(id);
  }, []);
  const remainingMs = Math.max(0, coolingExpiresAt - now);
  if (remainingMs === 0) return null;
  const HH = Math.floor(remainingMs / 3_600_000);
  const MM = Math.floor((remainingMs % 3_600_000) / 60_000);
  return (
    <div className="flex items-center gap-1.5 text-sm text-warning" aria-live="polite">
      <Clock className="size-4" />
      <span>{t('cooling.indicator', { HH: String(HH), MM: String(MM).padStart(2, '0') })}</span>
    </div>
  );
}
```

The 60-second interval matches typical cooling-window granularity (no need for sub-minute updates). `aria-live="polite"` on the wrapper announces minute-tick updates to screen readers.

import { getTranslations } from 'next-intl/server';
import type { Metadata } from 'next';
import { db } from '@/db';
import { users } from '@/db/schema/auth';
import { eq } from 'drizzle-orm';

import { MainContainer } from '@/components/layout/MainContainer';

export async function generateMetadata(): Promise<Metadata> {
  const t = await getTranslations('profile');
  return { title: t('title') };
}

export default async function ProfilePage() {
  const session = await auth();
  if (!session?.user) redirect('/login');
  const userId = (session.user as { id?: string }).id;
  if (!userId) redirect('/login');

  const t = await getTranslations('profile');

  const userRows = await db.select({
