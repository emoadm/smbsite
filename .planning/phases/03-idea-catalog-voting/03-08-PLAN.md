**File 1: src/lib/voting/anomaly.ts**

```typescript
import 'server-only';
import * as Sentry from '@sentry/nextjs';
import { sql, and, eq, gte } from 'drizzle-orm';
import type { db as DrizzleDb } from '@/db';
import { vote_events_log, vote_anomalies, type AnomalyTrigger } from '@/db/schema/voting';
import { addEmailJob } from '@/lib/email/queue';
import { logger } from '@/lib/logger';
import { COOLING_INTERVAL } from './cooling';

// Tunable thresholds (RESEARCH §1)
const PER_IDEA_N1 = Number(process.env.ANOMALY_PER_IDEA_N1 ?? 30);
const PER_IDEA_T1_MIN = Number(process.env.ANOMALY_PER_IDEA_T1_MIN ?? 10);
const SUBNET_N2 = Number(process.env.ANOMALY_SUBNET_N2 ?? 20);
const SUBNET_T2_MIN = Number(process.env.ANOMALY_SUBNET_T2_MIN ?? 30);
const FRESH_PCT = Number(process.env.ANOMALY_FRESH_PCT ?? 50);
const FRESH_WINDOW_N3 = Number(process.env.ANOMALY_FRESH_WINDOW_N3 ?? 20);

type Trigger = AnomalyTrigger;

export async function detectAnomaliesForIdea(db: typeof DrizzleDb, ideaId: string): Promise<void> {
  const [perIdea, subnet, fresh] = await Promise.all([
    checkPerIdeaVelocity(db, ideaId),
    checkSubnetCluster(db, ideaId),
    checkFreshAccountShare(db, ideaId),
  ]);
  for (const result of [perIdea, subnet, fresh]) {
    if (result.fired) {
      await emitAnomaly(db, ideaId, result.trigger, result.count, result.windowMin, result.freshPct);
    }
  }
}

async function checkPerIdeaVelocity(db: typeof DrizzleDb, ideaId: string) {
  const r = await db.execute(sql`
    SELECT COUNT(*)::int AS n
    FROM vote_events_log
    WHERE idea_id = ${ideaId}
      AND occurred_at > NOW() - INTERVAL '${sql.raw(`${PER_IDEA_T1_MIN} minutes`)}'
      AND action IN ('cast', 'change')
  `);
  const n = Number((r.rows[0] as { n: number }).n);
  return { fired: n > PER_IDEA_N1, trigger: 'per_idea_velocity' as const, count: n, windowMin: PER_IDEA_T1_MIN, freshPct: undefined as number | undefined };
}

async function checkSubnetCluster(db: typeof DrizzleDb, ideaId: string) {
  // Find the subnet_hash with the highest count in T2 minutes for this idea
  const r = await db.execute(sql`
    SELECT subnet_hash, COUNT(*)::int AS n
    FROM vote_events_log
    WHERE idea_id = ${ideaId}
      AND occurred_at > NOW() - INTERVAL '${sql.raw(`${SUBNET_T2_MIN} minutes`)}'
      AND action IN ('cast', 'change')
    GROUP BY subnet_hash
    ORDER BY n DESC
    LIMIT 1
  `);
  const top = r.rows[0] as { subnet_hash: string; n: number } | undefined;
  const n = Number(top?.n ?? 0);
  return { fired: n > SUBNET_N2, trigger: 'subnet_cluster' as const, count: n, windowMin: SUBNET_T2_MIN, freshPct: undefined };
}

async function checkFreshAccountShare(db: typeof DrizzleDb, ideaId: string) {
  // Of the last N3 votes for this idea, what % are fresh accounts?
  const r = await db.execute(sql`
    WITH recent AS (
      SELECT fresh_account_at_event
      FROM vote_events_log
      WHERE idea_id = ${ideaId}
        AND action IN ('cast', 'change')
      ORDER BY occurred_at DESC
      LIMIT ${FRESH_WINDOW_N3}
    )
    SELECT
      (CASE WHEN COUNT(*) > 0 THEN ROUND(100.0 * SUM(CASE WHEN fresh_account_at_event THEN 1 ELSE 0 END)::numeric / COUNT(*)) ELSE 0 END)::int AS pct,
      COUNT(*)::int AS n
    FROM recent
  `);
  const row = r.rows[0] as { pct: number; n: number };
  const pct = Number(row?.pct ?? 0);
  const n = Number(row?.n ?? 0);
  // Only fire if we have enough data (full window) AND pct exceeds threshold
  return { fired: n >= FRESH_WINDOW_N3 && pct > FRESH_PCT, trigger: 'fresh_account_share' as const, count: n, windowMin: undefined, freshPct: pct };
}

async function emitAnomaly(
  db: typeof DrizzleDb,
  ideaId: string,
  trigger: Trigger,
  count: number,
  windowMin?: number,
  freshPct?: number,
) {
  // UPSERT — keep one active anomaly per (idea, trigger). On re-fire within the same active row, bump last_detected_at + count.
  // Use raw SQL because Drizzle's upsert with WHERE on partial-index requires a constraint or a dedicated index.
  // Strategy: try INSERT first; if a row with status='unresolved' exists for this (idea, trigger), UPDATE.
  const existing = await db.execute(sql`
    SELECT id FROM vote_anomalies
    WHERE idea_id = ${ideaId} AND trigger_type = ${trigger} AND status = 'unresolved'
    LIMIT 1
  `);
  let anomalyId: string;
  if (existing.rows.length > 0) {
    anomalyId = (existing.rows[0] as { id: string }).id;
    await db.execute(sql`
      UPDATE vote_anomalies SET count = ${count}, last_detected_at = NOW()
      WHERE id = ${anomalyId}
    `);
  } else {
    const insert = await db.insert(vote_anomalies).values({
      idea_id: ideaId,
      trigger_type: trigger,
      count,
      status: 'unresolved',
    }).returning({ id: vote_anomalies.id });
    anomalyId = insert[0].id;
  }

  // Sentry — PII-free tags only (RESEARCH §11 / Phase 1 D-21)
  try {
    Sentry.captureMessage('vote_velocity_anomaly', {
      level: 'warning',
      tags: {
        idea_id: ideaId,
        trigger_type: trigger,
        count: String(count),
        ...(windowMin !== undefined ? { window_min: String(windowMin) } : {}),
        ...(freshPct !== undefined ? { fresh_pct: String(freshPct) } : {}),
      },
      contexts: { anomaly: { anomalyId } },
    });
  } catch (err) {
    logger.warn({ event: 'sentry_anomaly_emit_failed', err_kind: (err as Error).constructor.name }, 'sentry emit failed');
  }

  // Email — recipientEmail looked up below; fire-and-forget
  const recipient = process.env.EDITOR_NOTIFICATION_EMAIL ?? await firstAdminEmail(db);
  const ideaTitle = await ideaTitleFor(db, ideaId);
  if (recipient) {
    try {
      await addEmailJob({
        kind: 'vote-anomaly-alert',
        anomalyId,
        ideaId,
        triggerType: trigger,
        triggerCount: count,
        ideaTitle,
        recipientEmail: recipient,
      });
    } catch (err) {
      logger.warn({ event: 'anomaly_email_enqueue_failed', err_kind: (err as Error).constructor.name }, 'enqueue failed');
    }
  }

  logger.info({ event: 'vote_anomaly_emitted', anomaly_id: anomalyId, idea_id: ideaId, trigger_type: trigger, count }, 'anomaly emitted');
}

async function firstAdminEmail(db: typeof DrizzleDb): Promise<string | null> {
  const r = await db.execute(sql`SELECT email FROM users WHERE role = 'admin' ORDER BY created_at ASC LIMIT 1`);
  return (r.rows[0] as { email?: string })?.email ?? null;
}

async function ideaTitleFor(db: typeof DrizzleDb, ideaId: string): Promise<string> {
  const r = await db.execute(sql`SELECT title FROM ideas WHERE id = ${ideaId} LIMIT 1`);
  return (r.rows[0] as { title?: string })?.title ?? 'Идея';
}

// Email render+send (called from worker.ts)
import { render } from '@react-email/render';
import { VoteAnomalyAlert } from '@/emails/VoteAnomalyAlert';
import { sendBrevoEmail } from '@/lib/email/brevo'; // Phase 5 helper

export async function sendVoteAnomalyAlert(input: {
  anomalyId: string;
  ideaTitle: string;
  triggerType: Trigger;
  count: number;
  recipientEmail: string;
}): Promise<void> {
  const html = await render(VoteAnomalyAlert({
    ideaTitle: input.ideaTitle,
    triggerType: input.triggerType,
    count: input.count,
    anomalyId: input.anomalyId,
    adminUrl: `${process.env.AUTH_URL}/admin/views/vote-anomalies?anomaly_id=${input.anomalyId}`,
  }));
  const subject = `Аномалия в гласуване: ${input.ideaTitle}`;
  await sendBrevoEmail({
    to: input.recipientEmail,
    subject,
    html,
    headers: { 'X-Phase': '03-vote-anomaly' },
  });
  logger.info({ event: 'anomaly_alert_sent', anomaly_id: input.anomalyId, recipient_user_id: '<redacted>' }, 'alert email sent');
}

## Vote audit HMAC rotation (annual)

Cadence: every January (or sooner if compromise suspected).

1. Generate new secret: `openssl rand -hex 32`.
2. Set Fly.io secret: `flyctl secrets set VOTE_AUDIT_HMAC_SECRET=<new> -a smbsite-prod`.
   - Repeat for staging: `-a smbsite-staging`.
3. Set GitHub Actions secret: GitHub → Settings → Secrets → `VOTE_AUDIT_HMAC_SECRET`.
4. Bump epoch counter in this section (current: epoch 1 from 2026-05-07).
5. Document rotation event in `STATE.md` deferred items (resolved).

Rotation does NOT re-hash historic vote_events_log rows — that would defeat the GDPR
rationale (raw IPs are not retained). Per Phase 3 D-15: anomaly alerts fire on ≤30-min
windows, so cross-epoch comparison is never required for OPS-04 detection. Old hashes
remain matchable within their own epoch for forensic purposes; cross-epoch matching
requires external correlation, which is intentional.

If a leak of the OLD epoch's secret occurs: rotate immediately + notify DPO. Hashes
from the old epoch become rainbow-table-ble; the rows themselves are not GDPR-violating
(no raw IP) but the cluster-detection signal is degraded for the leaked epoch.

**Current epoch:** 1 (start: 2026-05-07; rotated: never)
```
