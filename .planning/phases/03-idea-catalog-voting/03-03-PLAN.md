```sql
-- Phase 3 — Payload Ideas collection schema
-- Generated by plan 03-03; manual application is REQUIRED because payload migrate
-- is BLOCKED on this project (project_payload_schema_constraint memory).
--
-- Apply order: staging first, verify, then production.
-- Verify after apply with: \d ideas

BEGIN;

CREATE TABLE IF NOT EXISTS ideas (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  title text NOT NULL,
  slug text NOT NULL UNIQUE,
  topic text NOT NULL,
  excerpt text,
  hero_id integer REFERENCES media(id) ON DELETE SET NULL,
  body jsonb NOT NULL,
  is_featured boolean NOT NULL DEFAULT false,
  featured_order integer,
  status text NOT NULL DEFAULT 'draft',
  display_frozen boolean NOT NULL DEFAULT false,
  created_at timestamptz NOT NULL DEFAULT NOW(),
  updated_at timestamptz NOT NULL DEFAULT NOW()
);

CREATE INDEX IF NOT EXISTS ideas_status_created_idx ON ideas(status, created_at DESC);
CREATE INDEX IF NOT EXISTS ideas_topic_idx ON ideas(topic);
CREATE INDEX IF NOT EXISTS ideas_featured_idx ON ideas(is_featured, featured_order, created_at DESC) WHERE is_featured = true;

-- Topic enum check constraint (text col over pgEnum, with CHECK for defense-in-depth):
ALTER TABLE ideas DROP CONSTRAINT IF EXISTS ideas_topic_check;
ALTER TABLE ideas ADD CONSTRAINT ideas_topic_check CHECK (topic IN ('taxes','labor','regulation','financing','digitalization','other'));

-- Status enum check constraint:
ALTER TABLE ideas DROP CONSTRAINT IF EXISTS ideas_status_check;
ALTER TABLE ideas ADD CONSTRAINT ideas_status_check CHECK (status IN ('draft','published','archived'));

-- Trigger to auto-update updated_at column (matches Phase 5 Newsletters pattern; if a
-- shared trigger function `set_updated_at()` already exists from a prior phase, skip the
-- CREATE FUNCTION block and just attach the trigger):
CREATE OR REPLACE FUNCTION set_updated_at() RETURNS trigger AS $$
BEGIN
  NEW.updated_at := NOW();
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

DROP TRIGGER IF EXISTS ideas_set_updated_at ON ideas;
CREATE TRIGGER ideas_set_updated_at
  BEFORE UPDATE ON ideas
  FOR EACH ROW EXECUTE FUNCTION set_updated_at();

COMMIT;

-- Verification queries (run after COMMIT):
-- \d ideas
-- SELECT indexname FROM pg_indexes WHERE tablename = 'ideas';
-- SELECT conname FROM pg_constraint WHERE conrelid = 'ideas'::regclass;
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Payload admin (editor session) → Postgres `ideas` table | Editor authors content; Payload validates fields server-side; row writes hit live DB |
| Public anonymous request → catalog page reading from Drizzle | Drizzle reads `ideas WHERE status='published'`; Payload REST is not the public read path |
| Operator session → Neon Console SQL Editor | Operator pastes DDL; misapplied DDL is unrecoverable without backup restore |

## STRIDE Threat Register

| Threat ID | Category | Component | Severity | Disposition | Mitigation Plan |
|-----------|----------|-----------|----------|-------------|-----------------|
| T-03-03-01 | EoP | Non-editor invokes Payload create/update on Ideas | HIGH | mitigate | `isEditorOrAdmin` access function on create/update/delete (Newsletters parity); plan 03-07 adds defense-in-depth `assertEditorOrAdmin` inside admin Server Actions |
| T-03-03-02 | Tampering | Editor changes slug after publish (link-rot) | MEDIUM | mitigate | beforeChange hook throws APIError when originalDoc.status === 'published' and slug differs |
| T-03-03-03 | Tampering | Editor inserts banned Lexical block (raw HTML, code) | HIGH | mitigate | Lexical features whitelist enforced in code; tests/unit/ideas-lexical-config.test.ts grep-asserts banned features absent; lexical-to-html renderer (Phase 5) only handles whitelisted node types |
| T-03-03-04 | DoS | Operator runs DDL twice causing duplicate constraint error | LOW | mitigate | DDL uses `IF NOT EXISTS` for table + indexes + `DROP CONSTRAINT IF EXISTS` before ADD; idempotent re-runs |

All HIGH+ threats are mitigated within this plan or in plan 03-07's Server Action layer.
</threat_model>
